\documentclass[tikz,border=5mm]{standalone}
\usepackage{amsmath}
\usetikzlibrary{arrows.meta, positioning, fit}

% Define styles for consistency
\tikzset{
    mc/.style={draw, rectangle, minimum width=1.2cm, minimum height=0.8cm, align=center},
    node distance=1.5cm,
    >=Stealth,
}

\begin{document}

\begin{tikzpicture}

% Upper row (Low resolution)
\node[mc] (mc1L) {MC};
\node[mc, right=of mc1L] (mc2L) {MC};
\node[mc, right=of mc2L] (mc3L) {MC};
\node[right=of mc3L] (dots1) {$\cdots$};
\node[mc, right=of dots1] (mc4L) {MC};

% Lower row (High resolution)
\node[mc, below=2.5cm of mc1L] (mc1H) {MC};
\node[mc, right=of mc1H] (mc2H) {MC};
\node[mc, right=of mc2H] (mc3H) {MC};
\node[right=of mc3H] (dots2) {$\cdots$};
\node[mc, right=of dots2] (mc4H) {MC};

% Labels for input voltages
\node[left=0.5cm of mc1L.west] (vL) {$V_\mathcal{L}$};
\node[left=0.5cm of mc1H.west] (vH) {$V_\mathcal{H}$};

% Arrows connecting input voltages to MC blocks
\draw[->] (vL) -- (mc1L);
\draw[->] (vH) -- (mc1H);

% Arrows within the upper row
\draw[->] (mc1L) -- node[above] {$i_{\mathcal{L}_1}$} (mc2L);
\draw[->] (mc2L) -- node[above] {$i_{\mathcal{L}_2}$} (mc3L);
\draw[->] (mc3L) -- (dots1);
\draw[->] (dots1) -- (mc4L);

% Arrows within the lower row
\draw[->] (mc1H) -- node[above] {$i_{\mathcal{H}_1}$} (mc2H);
\draw[->] (mc2H) -- node[above] {$i_{\mathcal{H}_2}$} (mc3H);
\draw[->] (mc3H) -- (dots2);
\draw[->] (dots2) -- (mc4H);

% Diagonal arrows between rows
\draw[->] (mc2L.south east) -- node[pos=0.6, above, sloped] {$i_{\text{V}_\mathcal{L}_2}$} (mc2H.north east);
\draw[->] (mc3L.south west) -- node[pos=0.6, above, sloped] {$i_{\text{V}_\mathcal{H}_2}$} (mc3H.north west);

% Output arrows from the last MC blocks
\draw[->] (mc4L.east) -- node[above] {$\hat{i}_{\text{V}_\mathcal{L}}$} +(2.5,0);
\draw[->] (mc4H.east) -- node[above] {$\hat{i}_{\text{V}_\mathcal{H}}$} +(2.5,0);

% Final output combining the two streams
\node at ($(mc4L.east)+(3,0)$) (add) {$\oplus$};
\node[right=0.8cm of add] (final_out) {$\hat{i}$};
\draw[->] ([xshift=-0.2cm]add.east) -- (final_out);

% Fit boxes around the rows
\node[fit=(mc1L) (mc4L), draw, inner sep=5pt] {};
\node[fit=(mc1H) (mc4H), draw, inner sep=5pt] {};

\end{tikzpicture}

\end{document}
