\documentclass[tikz,border=5mm]{standalone}
\usetikzlibrary{arrows.meta, decorations.pathreplacing}

\begin{document}
\begin{tikzpicture}[scale=1.5, every node/.style={font=\footnotesize}]

% Define coordinates
\coordinate (A) at (0,0); % Start of T_i
\coordinate (B) at (4,0); % End of T_i
\coordinate (C) at (5,1.5); % Start of T_{i+1}

% Draw T_i (horizontal line)
\draw[very thick] (A) -- (B) node[midway,below] {$\widetilde{T}_i$};

% Draw T_{i+1} (inclined line)
\draw[very thick] (C) -- ++(3,1.5) node[midway,right] {$\widetilde{T}_{i+1}$};

% Draw the arc C_i (dotted line)
\draw[dotted, thick] (B) to[bend left=45] node[pos=0.5, above] {$C_i$} (C);

% Mark the endpoints of the arc with dots
\fill (B) circle (1.5pt);
\fill (C) circle (1.5pt);

% Add the dotted segments along the arc
\foreach \i in {1,...,9}{ % 10 segments in total
    \pgfmathsetmacro{\pos}{\i/10} % Position along the path
    \coordinate (P\i) at ($(B)!{\pos}!(C)$);
    \fill (P\i) circle (0.5pt);
}
