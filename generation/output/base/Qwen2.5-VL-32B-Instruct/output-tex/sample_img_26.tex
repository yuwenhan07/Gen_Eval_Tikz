\documentclass[tikz,border=5mm]{standalone}
\usetikzlibrary{angles, quotes}

\begin{document}
\begin{tikzpicture}[scale=1.5, thick]

% Define parameters
\def\rmin{1.5} % Inner radius (s_min)
\def\rmax{2.5} % Outer radius (s_max)
\def\reff{2.0} % Effective radius (s_eff)
\def\thetamax{60} % Angular opening (θ_MAX)

% Draw the outer arc (s_max)
\draw[dashed] (0,0) ++(180-\thetamax:2.7) arc (180-\thetamax:180+\thetamax:2.7);

% Draw the inner arc (s_min)
\draw[dashed] (0,0) ++(180-\thetamax:\rmin) arc (180-\thetamax:180+\thetamax:\rmin);

% Draw the effective distance arc (s_eff)
\draw[dashed] (0,0) ++(180-\thetamax:\reff) arc (180-\thetamax:180+\thetamax:\reff);

% Draw the central vertical line
\draw[dashed] (0,\rmax) -- (0,-\rmax);

% Draw the shaded region between s_min and s_max
\fill[gray!20] (0,0) ++(180-\thetamax:\rmin) arc (180-\thetamax:180+\thetamax:\rmin) 
               -- (0,0) ++(180+\thetamax:\rmax) arc (180+\thetamax:180-\thetamax:\rmax) -- cycle;

% Draw the radial lines for s_min, s_max, and s_eff
\draw[->, thick] (0,0) -- (\thetamax:\rmax) node [right] {$s_{\text{max}}$};
\draw[->, thick] (0,0) -- (\thetamax:\reff) node [above right] {$s_{\text{eff}}$};
\draw[->, thick] (0,0) -- (\thetamax:\rmin) node [left] {$s_{\text{min}}$};

% Draw the angular opening (θ_MAX)
\pic [draw, ->, "$\theta_{\text{MAX}}$", angle eccentricity=1.5, angle radius=1cm] {angle = {(180+\thetamax:\rmax)--(0,0)--(180-\thetamax:\rmax)}};

% Label the origin
\node at (0,0) [below left] {${\bf O}$};

% Add dotted lines for guidance (optional)
\foreach \ang in {-\thetamax, 0, \thetamax}
    \draw[dotted, gray!50] (0,0) -- (\ang:\rmax);

\end{tikzpicture}
\end{document}
