\documentclass{standalone}
\usepackage{pgfplots}
\usepackage{xcolor}

% Define custom colors
\definecolor{arr2flatcolour}{HTML}{4F81BD} % Blue
\definecolor{minnesotacolour}{HTML}{FFC000} % Yellow
\definecolor{gaussiancolour}{HTML}{37A2DA} % Green
\definecolor{rhscolour}{HTML}{91E8E1} % Purple
\definecolor{arr2minncolour}{HTML}{D65F5F} % Red
\definecolor{arr2bumpcolour}{HTML}{6CA0DC} % Light Blue

\begin{document}

\begin{tikzpicture}
    \begin{axis}[
        name=plot1,
        width=12cm, height=6cm,
        xlabel={$R^2$},
        ylabel={Prior density},
        xmin=0, xmax=1.05,
        ymin=0, ymax=1.2,
        xtick={0,0.25,...,1},
        ytick=\empty,
        enlargelimits=false,
        axis lines=left,
        ticklabel style={font=\footnotesize},
        legend pos=north west,
        legend cell align=left,
        legend style={nodes={scale=0.8, transform shape}},
        every axis plot post/.append style={thick}
    ]

    % RHS prior
    \addplot [smooth, domain=0:1, color=rhscolour]
        {exp(-1/(1-x))};
    \addlegendentry{RHS};

    % Minnesota prior
    \addplot [smooth, domain=0:1, color=minnesotacolour]
        {1/sqrt(2*pi*x)*(exp(-(log(x))^2/2)/x)};
    \addlegendentry{Minn.};

    % ARR2 prior
    \addplot [smooth, domain=0:1, color=arr2flatcolour]
        {1};
    \addlegendentry{ARR2};

    % Independent Gaussian priors
    \addplot [smooth, domain=0:1, color=gaussiancolour]
        {exp(-5*(x-0.8)^2)};
    \addlegendentry{Gauss.};

    \end{axis}

    % Relative R^2 contribution plots
    \begin{scope}[shift={(0,-7cm)}]
        \begin{axis}[
            name=plot2,
            width=6cm, height=5cm,
            xlabel={$\phi_i$},
            ylabel={Relative $R^2$},
            xmin=0.5, xmax=12.5,
            ymin=-0.05, ymax=0.55,
            xtick={1,2,...,12},
            ytick={0,0.2,0.4},
            enlargelimits=false,
            axis lines=left,
            ticklabel style={font=\footnotesize},
            legend pos=south west,
            legend cell align=left,
            legend style={nodes={scale=0.8, transform shape}},
            every axis plot post/.append style={thick}
        ]

        % Minnesota-type prior
        \addplot [smooth, domain=1:12, color=arr2minncolour]
            {0.5*exp(-((x-2)/2)^2)};
        \addlegendentry{ARR2 (Minn.)};

        % Gaussian prior
        \addplot [smooth, domain=1:12, color=gaussiancolour]
            {0.1*exp(-((x-2)/2)^2)};
        \addlegendentry{ARR2 (Bump)};

        % Flat prior
        \addplot [smooth, domain=1:12, color=arr2flatcolour]
            {0.1};
        \addlegendentry{ARR2 (Flat)};

        \end{axis}

        % Second relative R^2 contribution plot
        \begin{axis}[
            at={(plot2.east), xshift=1.5cm},
            anchor=west,
            width=6cm, height=5cm,
            xlabel={$\phi_i$},
            ylabel={Relative $R^2$},
            xmin=0.5, xmax=12.5,
            ymin=-0.05, ymax=0.55,
            xtick={1,2,...,12},
            ytick={0,0.2,0.4},
            enlargelimits=false,
            axis lines=left,
            ticklabel style={font=\footnotesize},
            legend pos=south west,
            legend cell align=left,
            legend style={nodes={scale=0.8, transform shape}},
            every axis plot post/.append style={thick}
        ]

        % Minnesota-type prior
        \addplot [smooth, domain=1:12, color=arr2minncolour]
            {0.5*exp(-((x-2)/2)^2)};
        \addlegendentry{ARR2 (Minn.)};

        % Bump prior
        \addplot [smooth, domain=1:12, color=arr2bumpcolour]
            {0.1*exp(-((x-2)/2)^2)};
        \addlegendentry{ARR2 (Bump)};

        % Flat prior
        \addplot [smooth, domain=1:12, color=arr2flatcolour]
            {0.1};
        \addlegendentry{ARR2 (Flat)};

        \end{axis}
    \end{scope}
\end{tikzpicture}

\end{document}
