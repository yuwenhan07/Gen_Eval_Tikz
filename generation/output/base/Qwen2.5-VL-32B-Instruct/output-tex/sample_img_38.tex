\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, positioning}

\begin{document}

\begin{tikzpicture}[
    node distance=2cm,
    every node/.style={circle, draw, minimum width=2cm, font=\sffamily},
    >=Latex % Set arrow style
]

% Define nodes
\node (type) {Type};
\node (size) [below left=of type] {Size};
\node (weight) [below=of size] {Weight};
\node (gripper) [below right=of weight] {Gripper};
\node (robot) [below=of gripper] {Robot};
\node (texture) [below right=of type] {Texture};
\node (target) [below right=of texture] {Target};
\node (rigidity) [right=of texture] {Rigidity};
\node (goal) [below=of target] {Goal};
\node (success) [below right=of target] {Success};

% Draw directed edges
\draw[->] (type) -- (size);
\draw[->] (type) -- (weight);
\draw[->] (type) -- (texture);
\draw[->] (type) -- (rigidity);

\draw[->] (size) -- (gripper);
\draw[->] (weight) -- (gripper);
\draw[->] (texture) -- (gripper);
\draw[->] (rigidity) -- (gripper);

\draw[->] (gripper) -- (target);
\draw[->] (robot) -- (gripper);

\draw[->] (target) -- (success);
\draw[->] (goal) -- (success);

\end{tikzpicture}

\end{document}
