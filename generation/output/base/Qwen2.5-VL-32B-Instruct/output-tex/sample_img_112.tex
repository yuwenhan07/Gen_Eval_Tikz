\documentclass[tikz,border=10pt]{standalone}
\usetikzlibrary{arrows.meta, bending}

\begin{document}
\begin{tikzpicture}[
    scale=1.5,
    >=Latex,
    every node/.style={circle, fill=black, inner sep=2pt},
    state/.style={circle, draw=white, minimum size=4mm, inner sep=0pt},
    dashed edge/.style={red, dashed, thick},
    solid edge/.style={blue, thick},
    dotted edge/.style={green!70!black, dotted, thick},
    every label/.style={black, font=\small},
]

% Define the positions of the states
\coordinate (v) at (0, 2);
\coordinate (u) at (-2, 1);
\coordinate (s1) at (-1.5, 0.5);
\coordinate (s2) at (-1, 0);
\coordinate (s3) at (0, -0.5);
\coordinate (s4) at (1, 0);
\coordinate (s5) at (1.5, 0.5);
\coordinate (s6) at (0, 1);

% Draw the states
\node[state] at (v) [label=above:$v$] {};
\node[state] at (u) [label=left:$u$] {};
\node[state] at (s1) {};
\node[state] at (s2) {};
\node[state] at (s3) {};
\node[state] at (s4) {};
\node[state] at (s5) {};
\node[state] at (s6) {};

% Draw the edges
\draw[dashed edge, ->]
    (v) to[bend right=15] node[midway, left, sloped] {\tiny $-$} (u);
\draw[solid edge, ->]
    (v) to[bend left=15] node[midway, right, sloped] {\tiny $+$} (s1);
\draw[solid edge, ->]
    (u) to[bend left=15] node[midway, above, sloped] {\tiny $+$} (s1);
\draw[dotted edge, ->]
    (s1) to[bend right=15] node[midway, left, sloped] {\tiny $-$} (s2);
\draw[solid edge, ->]
    (s2) to[bend right=15] node[midway, below, sloped] {\tiny $+$} (s3);
\draw[dashed edge, ->]
    (s3) to[bend right=15] node[midway, right, sloped] {\tiny $-$} (s4);
\draw[solid edge, ->]
    (s4) to[bend right=15] node[midway, right, sloped] {\tiny $+$} (s5);
\draw[dotted edge, ->]
    (s5) to[bend right=15] node[midway, right, sloped] {\tiny $-$} (s6);
\draw[solid edge, ->]
    (s6) to[bend left=15] node[midway, above, sloped] {\tiny $+$} (v);
\draw[dotted edge, ->]
    (s3) to[bend left=15] node[midway, above, sloped] {\tiny $-$} (u);
\draw[solid edge, ->]
    (s5) to[bend left=15] node[midway, above, sloped] {\tiny $+$} (s1);

\end{tikzpicture}
\end{document}
