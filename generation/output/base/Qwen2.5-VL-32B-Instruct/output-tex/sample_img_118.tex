\documentclass[tikz,border=10pt]{standalone}
\usepackage{amssymb}

\begin{document}
\begin{tikzpicture}[scale=1.5]

% Define coordinates for the regular pentagon K
\def\r{2} % Radius of the circumscribed circle
\foreach \i in {1,...,5} {
    \coordinate (v\i) at ({90+72*(\i-1)}:\r);
}

% Draw K (solid black pentagon)
\draw[thick, black] (v1) -- (v2) -- (v3) -- (v4) -- (v5) -- cycle;

% Draw -K (dashed black pentagon, reflected about origin)
\draw[thick, dashed, black] (v1) -- (v2) -- (v3) -- (v4) -- (v5) -- cycle;
\foreach \i in {1,...,5} {
    \coordinate (-v\i) at ({90+72*(\i-1)+180}:\r);
}
\draw[thick, dashed, black] (-v1) -- (-v2) -- (-v3) -- (-v4) -- (-v5) -- cycle;

% Draw \underline{M}_0(K,-K) (orange quadrilateral)
\coordinate (m0a) at ($(v1)!0.5!(v2)$);
\coordinate (m0b) at ($(v2)!0.5!(v3)$);
\coordinate (m0c) at ($(v4)!0.5!(v5)$);
\coordinate (m0d) at ($(v5)!0.5!(v1)$);
\draw[thick, orange] (m0a) -- (m0b) -- (m0c) -- (m0d) -- cycle;

% Draw \overline{M}_1(K,-K) (blue ellipse)
\draw[thick, blue] (0,0) circle (1.8);

% Draw \overline{M}_3(K,-K) (red ellipse)
\draw[thick, red] (0,0) circle (2.2);

% Draw \overline{M}_\infty(K,-K) (dashdotted circle)
\draw[thick, dashdotted] (0,0) circle (\r);

% Mark the origin
\node[circle, fill=black, inner sep=1.5pt, label={below:$0$}] at (0,0) {};

% Add annotations (arrows pointing to boundary points)
\draw[-latex, thick] (0.75*\r,0.75*\r) -- (0.75*\r+0.25,0.75*\r+0.25) node[right] {$\overline{M}_\infty(K,-K)$};
\draw[-latex, thick] (1.6,0) -- (1.8,0) node[right] {$\overline{M}_3(K,-K)$};
\draw[-latex, thick] (1.3,0) -- (1.5,0) node[right] {$\overline{M}_1(K,-K)$};
\draw[-latex, thick] (0.8,0.8) -- (1.0,1.0) node[right] {$\underline{M}_0(K,-K)$};

\end{tikzpicture}
\end{document}
