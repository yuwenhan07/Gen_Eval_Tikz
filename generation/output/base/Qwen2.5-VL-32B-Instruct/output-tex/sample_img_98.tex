\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{decorations.pathmorphing} % For wavy lines

\begin{document}

\begin{tikzpicture}[scale=1.2, every node/.style={font=\footnotesize}]

% Define some parameters for positioning and sizing
\def\MassWidth{1.5}
\def\MassHeight{1.0}
\def\SmallMassWidth{0.6}
\def\SmallMassHeight{0.6}
\def\SpringLength{3.0}
\def\ArrowLength{0.4}
\def.labelXOffset{-0.2}
\def\labelYOffset{-0.2}
\def\dashedLineLength{1.5}

% Draw the large mass M
\filldraw[fill=blue!20, draw=black]
  (0, 0) rectangle (\MassWidth, \MassHeight)
  node[midway] {$M$};

% Draw the small mass m
\filldraw[fill=blue!20, draw=black]
  (\SpringLength+\MassWidth+0.5*\SmallMassWidth, 0) rectangle 
  (\SpringLength+\MassWidth+\SmallMassWidth, \SmallMassHeight)
  node[midway] {$m$};

% Draw the spring connecting M to the wall
\draw [thick, decorate, decoration={zigzag, segment length=4pt, amplitude=1.5pt}]
  (-\SpringLength/2, 0.5*\MassHeight) -- (0, 0.5*\MassHeight);
\node at (-0.8*\SpringLength, 0.5*\MassHeight) [above right] {$K$};
\draw [thick] (-\SpringLength-0.5, 0.5*\MassHeight) -- (-\SpringLength-0.5, -0.5*\MassHeight); % Vertical line
\draw [thick] (-\SpringLength-0.5, 0.5*\MassHeight) -- (-\SpringLength-1, 0.5*\MassHeight); % Horizontal extension
\draw [thick] (-\SpringLength-0.5, -0.5*\MassHeight) -- (-\SpringLength-1, -0.5*\MassHeight); % Horizontal extension

% Draw the spring connecting M to m
\draw [thick, decorate, decoration={zigzag, segment length=4pt, amplitude=1.5pt}]
  (\MassWidth, 0.5*\MassHeight) -- (\SpringLength+\MassWidth, 0.5*\SmallMassHeight);
\node at (\MassWidth+0.6*\SpringLength, 0.5*\MassHeight) [right] {$k$};

% Draw the x_1 displacement arrow
\draw [-latex, thick]
  (\MassWidth, -0.2) -- (\SpringLength+\MassWidth, -0.2)
  node [midway, below] {$x_1$};

% Draw the x_2 displacement arrow
\draw [-latex, thick]
  (\MassWidth, -0.6) -- (\SpringLength+\MassWidth, -0.6)
  node [midway, below] {$x_2$};

% Draw the L distance arrow
\draw [-latex, thick]
  (\MassWidth, -0.8) --++ (\SpringLength+0.5*\SmallMassWidth, 0)
  node [pos=0.9, above] {$L$};

% Draw the vertical dashed lines
\draw [dotted, thin]
  (\SpringLength+\MassWidth+\SmallMassWidth/2, 0) --++ (0, \SmallMassHeight+1);
\draw [dotted, thin]
  (\SpringLength+\MassWidth+\SmallMassWidth/2, 0) --++ (0, -1);

% Draw the velocity/gravitational force arrow (v, g)
\draw [-latex, red, thick]
  (\SpringLength+\MassWidth+\SmallMassWidth/2, \SmallMassHeight+0.5) --
  ++(-1.5, 0)
  node [pos=0.7, above left] {$v, g$};

\end{tikzpicture}

\end{document}
