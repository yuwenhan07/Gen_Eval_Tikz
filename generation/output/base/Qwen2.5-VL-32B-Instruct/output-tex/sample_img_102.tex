\documentclass{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.17}

\begin{document}

\begin{tikzpicture}
    \begin{axis}[
        width=\linewidth,
        height=0.8\linewidth,
        xmin=0, xmax=1,
        ymin=0, ymax=1,
        xlabel={Quantiles},
        ylabel={Cardiovascular Risk},
        grid=major,
        legend style={at={(0.98,0.5)}, anchor=east, nodes={scale=0.8, transform shape}},
        every axis plot post/.append style={line width=1.2pt},
        ticklabel style={font=\footnotesize},
        label style={font=\small},
        xtick distance=0.2,
        ytick distance=0.2,
        samples=100,
        smooth
    ]

        % Factual y' (Labels)
        \addplot [color=black, mark=none] {x^2};
        \addlegendentry{Factual $y'$ (Labels)}

        % Target y* (Labels)
        \addplot [color=red, mark=none] {0.5*x + 0.3};
        \addlegendentry{Target $y^*$ (Labels)}

        % Counterfactual y (MLPNet)
        \addplot [color=orange, mark=none] {sin(2*pi*deg(x)) / 4 + 0.6};
        \addlegendentry{Counterfactual $y$ (MLPNet)}

        % Counterfactual y (SVM)
        \addplot [color=blue, mark=none] {x^3 - 0.2*x + 0.5};
        \addlegendentry{Counterfactual $y$ (SVM)}

        % Counterfactual y (RBFNet)
        \addplot [color=green, mark=none] {exp(-5*(x-0.8)^2) + 0.3*x};
        \addlegendentry{Counterfactual $y$ (RBFNet)}

        % Vertical line at 0.6
        \draw [very thick, red] (axis cs:0.6,0) -- (axis cs:0.6,1);
    \end{axis}
\end{tikzpicture}

\end{document}
