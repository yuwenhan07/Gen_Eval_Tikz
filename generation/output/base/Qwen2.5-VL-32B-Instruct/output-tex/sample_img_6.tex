\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{calc}

\begin{document}
\begin{tikzpicture}[scale=0.8, every node/.style={inner sep=1.5pt}]

% Define the coordinates for the vertices
\foreach \i in {1,...,6} {
    \foreach \j in {1,...,9} {
        \pgfmathsetmacro{\x}{2*\i}
        \pgfmathsetmacro{\y}{-2*\j+1}
        \coordinate (v-\i-\j) at (\x,\y);
    }
}

% Draw the vertices
\foreach \i in {1,...,6} {
    \foreach \j in {1,...,9} {
        \filldraw[black] (v-\i-\j) circle (2pt);
    }
}

% Draw the horizontal lines (edges within each K_9)
\foreach \i in {1,...,6} {
    \foreach \j [evaluate=\j as \jj using int(\j+1)] in {1,...,8} {
        \draw (v-\i-\j) -- (v-\i-\jj);
    }
}

% Draw the vertical lines (edges between K_9 layers)
\foreach \i [evaluate=\i as \ii using int(\i+1)] in {1,...,5} {
    \foreach \j in {1,...,9} {
        \draw (v-\i-\j) -- (v-\ii-\j);
    }
}

% Draw the diagonal lines (edges connecting specific vertices across layers)
\foreach \i in {1,...,5} {
    \foreach \j in {1,...,9} {
        \pgfmathtruncatemacro{\k}{mod(\j+\i-1,9)+1}
        \draw (v-\i-\j) -- (v-\the\numexpr\i+1\relax-\k);
    }
}

% Draw the outer circles (for clarity)
\foreach \i in {1,...,6} {
    \foreach \j in {1,...,9} {
        \draw (v-\i-\j) circle (3pt);
    }
}

% Add labels for the rows
\foreach \j [evaluate=\j as \label using {x_{\j}}] in {1,...,9} {
    \node at (-1,-2*\j+1) {\(\label\)};
}

% Add labels for the columns
\foreach \i [evaluate=\i as \label using {\i K_9}] in {1,...,6} {
    \node at (2*\i,-10) {\(\label\)};
}

% Add numbers at the bottom
\foreach \i [evaluate=\i as \label using {\i}] in {1,...,6} {
    \node at (2*\i,-12) {\(\label\)};
}

% Draw the horizontal and vertical lines for the labels
\draw (-1,-19) -- (-1,-1) -- (13,-1) -- (13,-19);

\end{tikzpicture}
\end{document}
