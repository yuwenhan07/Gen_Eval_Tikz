\documentclass[border=5pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta, intersections}
\usepackage{amsmath}

\begin{document}
\begin{tikzpicture}[
  x=2.5cm, 
  y=2.5cm, 
  >=Stealth,
  font=\small
]
  % 定义颜色
  \colorlet{purpleline}{purple}
  \colorlet{tealline}{teal}
  \colorlet{greenline}{green!60!black}
  
  % 绘制坐标轴
  \draw[axisline] (0,0) -- (4.2,0) node[right] {$|v|$};
  \draw[axisline] (0,0) -- (0,1.2);
  
  % 标记y轴刻度
  \foreach \y/\label in {0/0, 1/1} {
    \draw (\smalltick,\y) -- +(-\smalltick,0) node[left] {$\label$};
  }
  
  % 标记x轴刻度点
  \foreach \x/\label in {0/t_0, 2/{\frac{\pi}{2}}, 2.5/{\frac{\pi}{2}+t_0}, 3/{\frac{\pi}{2}-t_0}, 3.5/{1-\tilde{\eta}}, 4/t'} {
    \draw (\x,\smalldash) -- + (0,-\smalldash) node[below] {$\label$};
  }
  
  % 绘制垂直虚线
  \foreach \x in {0, 2, 2.5, 3, 3.5, 4} {
    \draw[dashedline] (\x,0) -- (\x,1);
  }
  
  % 绘制紫色线段
  \draw[dataline, purpleline] (0,1) -- (2,0);
  \draw[dataline, purpleline] (2,0) -- (2.5,0);
  
  % 绘制蓝绿色和绿色线段
  \draw[dataline, tealline] (2.5,0) -- (3,1);
  \draw[dataline, tealline] (3,1) -- (4,1);
  \draw[dataline, greenline] (2.5,1) -- (3,0);
  \draw[dataline, greenline] (3,0) -- (4,0);
  
  % 添加标签
  \node[purpleline] at (0,1) {$\bullet$};
  \node[purpleline] at (2,0) {$\bullet$};
  \node[tealline] at (3,1) {$\bullet$};
  \node[greenline] at (3,0) {$\bullet$};
  
  % 添加额外文本标注
  \node[purpleline] at (1,1) {$\eta$};
  \node[tealline] at (3.75,1) {$1-\tilde{\eta}$};
\end{tikzpicture}
\end{document}
