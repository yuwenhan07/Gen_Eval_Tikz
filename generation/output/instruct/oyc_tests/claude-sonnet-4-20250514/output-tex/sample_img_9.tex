\tikzset{
    mcbox/.style={rectangle, draw=black, fill=white, minimum width=1cm, minimum height=0.8cm, thick},

    crossing/.style={thick},
    sumnode/.style={circle, draw=black, fill=white, minimum size=0.8cm, thick}
}

\begin{tikzpicture}[node distance=1.5cm]

% Top chain (VL)
\node[mcbox] (mc1_top) {MC};
\node[mcbox, right of=mc1_top] (mc2_top) {MC};

\node[mcbox, below=2cm of mc1_top] (mc1_bottom) {MC};

% Intermediate labels for top chain
\node[above=0.3cm of mc1_top] (il_label1) {$i_\mathcal{L}$};

\node[above left=0.2cm of mc2_bottom] (il_cross2) {$\dot{i}_\mathcal{V}_{\mathcal{L}_2}$};
\node[below right=0.2cm of mc1_top] (ih_cross1) {$i_{\mathcal{H}_1}$};
\node[below right=0.2cm of mc2_top] (ih_cross2) {$\dot{i}_\mathcal{V}_{\mathcal{H}_2}$};
