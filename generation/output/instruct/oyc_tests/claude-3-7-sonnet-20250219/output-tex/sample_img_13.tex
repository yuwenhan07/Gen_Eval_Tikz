\documentclass[tikz,border=5mm]{standalone}
\usetikzlibrary{arrows.meta}

\begin{document}
\begin{tikzpicture}[scale=0.8, 
    >=Latex,
    every node/.style={font=\footnotesize},
    myarrow/.style args={#1/#2}{draw=#2, -{Triangle[length=4pt]}, thick},
]

% Left part: Vertex labels
\begin{scope}[local bounding box=left]
  \draw[thick] (0,0) -- ++(1.5,0) coordinate[midway] (a)
               (0,0) -- ++(0,-1.5) coordinate[midway] (b)
               (0,0) -- ++(-1.5,0) coordinate[midway] (c)
               (0,0) -- ++(0,1.5) coordinate[midway] (d);
  
  \node[left] at (c) {$i_v$};
  \node[right] at (a) {$j_v$};
  \node[below] at (b) {$a_v$};
  \node[above] at (d) {$b_v$};
\end{scope}

% Right part: Stochastic six-vertex model
\begin{scope}[xshift=6cm, local bounding box=right]
  % Grid
  \draw[gray!50, thin] (-1,-3) grid (4,2);
  
  % Left boundary labels
  \foreach \y/\col/\num in {-3/-/-(0,-3),
                            -2/red/$j_{(-2)}=-\infty$,
                            -1/blue/$j_{(-1)}=-2$,
                             0/orange/$j_{(0)}=-\infty$,
                             1/green/$j_{(1)}=-1$,
                             2/cyan/$j_{(2)}=0$}
  {
    \fill[\col] (-1,\y) circle (2pt);
    \node[left=2pt] at (-1,\y) {\num};
  }
  
  % Arrows
  \draw[myarrow=0/cyan] (0,2) -- (4,2);
  \draw[myarrow=1/green] (0,1) -- (2,1);
  \draw[myarrow=2/orange] (2,1) -- (4,1);
  \draw[myarrow=3/blue] (0,0) -- (3,0);
  \draw[myarrow=4/orange] (3,0) -- (4,0);
  \draw[myarrow=5/red] (0,-1) -- (1,-1);
  \draw[myarrow=6/blue] (1,-1) -- (4,-1);
  \draw[myarrow=7/red] (0,-2) -- (4,-2);
  \draw[myarrow=8/cyan] (0,-3) -- (4,-3);
\end{scope}

\end{tikzpicture}
\end{document}
