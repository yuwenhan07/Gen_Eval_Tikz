\documentclass[tikz,border=3mm]{standalone}
\begin{document}
\foreach \X in {0,...,7} {
    \ifnum\X=0
        \tikzset{every node/.append style={draw}}
    \else
        \tikzset{every node/.append style={draw,inner sep=0pt,outer sep=0pt,anchor=center}}
    \fi
    \begin{tikzpicture}[rotate=90,scale=1.3]
        \pgfmathtruncatemacro{\N}{8} % Number of vertices
        \foreach \V in {1,...,\N}{
            \pgfmathtruncatemacro{\M}{mod(\V+\X,\N)+1}
            \node (\V) at ({(\M-1)/\N*pi}:1) {$\alpha_{-\M}$};
        }
        \path[draw,-stealth,line width=0.75pt]
            \foreach \V[count=\Vc] in {1,...,\N} 
                {\ifnum\V>c
                    (\V-1) edge[bend right] (\V)
                \fi}
            ;
        \foreach \V/\N in {1/0,2/-5,...,6/-1}{
            \draw (\V) -- ++(0,-1);
            \path (\V) ++(0,-1) node {$+\V$};
            \draw[|-|] (\V) ++(-1,0) -- ++(2*\N,0);
            \path (\V) ++(2*\N,0) node {\scriptsize $(\N\;\alpha\;-\N)$};
        }  
        \path[draw,-stealth,line width=0.75pt]
            \foreach \V[count=\Vc] in {1,...,\N} 
                {\ifnum\V>c
                    (\V-1) edge[bend left] (\V)
                \fi}
            ;
    \end{tikzpicture}
}
