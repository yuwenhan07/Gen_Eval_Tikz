\documentclass[tikz,border=3mm]{standalone}

\pgfplotsset{compat=1.16}

\tikzset{
    declare function={
        random(\xmin,\xmax,\ymin,\ymax)=
            (\xmin+\xmax)/2+0.5*rnd*(\xmax-\xmin)+rnd*(\ymax-\ymin);
    },
    declare function={
        minvalue(\xmin,\xmax,\ymin,\ymax)=min(random(\xmin,\xmax,\ymin,\ymax),\xmin,\xmax);
    },
    declare function={
        maxvalue(\xmin,\xmax,\ymin,\ymax)=max(random(\xmin,\xmax,\ymin,\ymax),\xmin,\xmax);
    },
    myplot/.style={draw,dashed,very thin},
    myplotcol/.style={myplot,#1},
    minima/.style={very thick},
    label/.style={align=center,fill=white,font=\small},
    }
