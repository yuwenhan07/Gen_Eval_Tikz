\documentclass[tikz,border=7mm]{standalone}
\begin{document}
\begin{tikzpicture}[scale=1.5]
  \def\R{sqrt(3)}
  \def\s{90}
  \def\t{60}
  \pgfmathsetmacro{\tangent}{atan(tan(\s)*(tan(\s)/tan(\t)) + sqrt(1 - tan(\s)*tan(\s)))}
  \draw (0:\R) arc[start angle=0,end angle=\s,radius=\R];
  \draw[dashed] (\tangent:\R) arc[start angle=\tangent,end angle=-\s,radius=\R];
  \draw[dashed] (0:-\R) arc[start angle=0,end angle=-\s,radius=\R];
  \draw (-\s:\R) -- (0,0) -- (-\tangent:\R);
  \draw[orange,dashed] (-\tangent:\R) -- ++({sin(\s)},-{cos(\s)}) coordinate[midway](tangent);
  \draw[orange,->] (tangent) -- ++({sin(\tangent+90)},{cos(\tangent+90)});
  \node[right] at (tangent){$t$};
  \node[right] at ({-\s/2+\tangent},{cos(\tangent)}){$s$};
\end{tikzpicture}
\end{document}
