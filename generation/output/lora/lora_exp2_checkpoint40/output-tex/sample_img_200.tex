\usepackage{tikz}
\usetikzlibrary{automata,arrows.meta}
\tikzset{
    >=Stealth,
    thick,
    scale=0.5,
    auto,
    every state/.style={
        draw=black,
        fill=black!10,
        minimum size=2cm,
        inner sep=0pt,
        outer sep=0pt,
    },
    every edge/.style={
        draw=black,
        line width=0.2mm,
    },
}
\begin{document}
\begin{tikzpicture}[node distance=2cm]
    \node[state] (A) {$s_1$};
    \node[state] (B)[right of=A] {$s_2$};
    \node[state] (C)[right of=B] {$s_3$};
    
    \path [-stealth] 
        (A) edge[loop above] node[left]{$1-p$, $2$} ()
        (A) edge node[midway,below,sloped] {$p$, $1$} (B)
        (B) edge[bend left=20] node[midway,above,sloped] {$1$, $0$} (C)
        (B) edge[bend right=20] node[midway,below,sloped] {$p$, $0$} (C)
        (C) edge[bend right=20] node[midway,below,sloped] {$1-p$, $1$} (A);
\end{tikzpicture}
