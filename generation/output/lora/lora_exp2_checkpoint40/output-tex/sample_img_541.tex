\documentclass{standalone}
\usepackage{amssymb}
\usepackage{tikz}
\usetikzlibrary{calc}
\tikzset{
    pics/sliced/.style n args={5}{
        code={
            \path[pic actions] (#1) coordinate (A) -- (#2) coordinate (B) -- (#3) coordinate (C) -- (#4) coordinate (D) -- cycle;
            \draw pic (#1,#2,#3,#4,"#5");
            \coordinate (#1,#2,#3,#4,#5) at ($(#1)!0.5!(#5)$);
        }
    },
    pics/sliced/.default=none none none none none,
    pics/sliced/.style args={#1,#2,#3,#4,#5}{
        code={
            \draw[dashed,#1] (#3,#5) -- (#4,#5);
            \draw[#2] (#1,#5) -- (#2,#5);
            \draw[dashed,#3] (#2,#4) -- (#4,#4);
            \draw[#4] (#2,#3) -- (#2,#4);
            \filldraw[fill=#5,#5] (#1,#5) circle (1pt) (#2,#5) circle (1pt) (#2,#3) circle (1pt) (#4,#4) circle (1pt);
        }
    }
}

\begin{document}
\begin{tikzpicture}
    \draw[sliced={none,none,none,none,black},fill=white] 
        (-1,-1) coordinate (A) -- (0,1) coordinate (B) -- (1,-1) coordinate (C) -- (0,-1) coordinate (D) -- cycle;
    \draw[sliced={none,none,none,none,black},fill=white] 
        (A) -- (B) -- ($(B)!1/3!(C)$) coordinate (E) -- (C) coordinate (F);
    \draw[sliced={none,none,none,none,black},fill=white] 
        ($(B)!2/3!(C)$) coordinate (G) -- (C) -- (F) -- ($(D)!1/3!(F)$) coordinate (H) -- ($(D)!2/3!(F)$) coordinate (I) -- (D);
    \draw[sliced={none,none,none,none,black},fill=white] 
        (E) -- (F) -- (G) coordinate (J) -- ($(G)!1/3!(H)$) coordinate (K) -- ($(G)!2/3!(H)$) coordinate (L) -- (H);
    \draw[sliced={none,none,none,none,black},fill=white] 
        (K) -- (L) -- ($(L)!1/3!(I)$) coordinate (M) -- ($(L)!2/3!(I)$) coordinate (N) -- (I) -- (H) coordinate (O);

    \draw[sliced={none,none,none,none,red}] 
        ($(H)!1/3!(O)$) coordinate (P) -- ($(I)!2/3!(O)$) coordinate (Q) -- ($(I)!1/3!(O)$) coordinate (R) -- ($(H)!2/3!(O)$) coordinate (S) -- (O);
    \draw[sliced={none,none,none,none,blue}] 
        ($(H)!1/3!(O)$) coordinate (T) -- ($(I)!2/3!(O)$) coordinate (U) -- ($(I)!1/3!(O)$) coordinate (V) -- ($(H)!2/3!(O)$) coordinate (W) -- (O);
    \foreach \i/\j in {A/E,C/J,H/P,I/Q,O/S,R/T,U/W,V}
    {
        \node [right] at (\i)--(\j){\i--\j};
    }
\end{tikzpicture}
\end{document}
