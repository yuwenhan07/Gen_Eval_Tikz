\begin{tikzpicture}[font=\sffamily]
    \def\r{0.7mm}
    \def\d{3}
    \def\s{0.1cm}
    \def\n{6}
    \def\two{2}

    \draw [->, line width=0.4mm] (-1,0)--++(180:\d) node [pos=.5, right] {Key};
    \draw [->, line width=0.4mm] (0,-1)--++(90:\d) node [pos=.5, above] {Time};

    \matrix(m)[matrix of nodes,
        nodes={minimum size=\s},
        column sep={\two*\s},
        row sep={\two*\s},
        inner sep=0pt,
        ]
    {
        &&&&&&\\
        &&&&&&\\
        &&&&&&\\
        &&&&&&\\
        &&&&&&\\
        &&&&&&\\
    };

    \foreach \j in {1,...,\n}{
        \pgfmathtruncatemacro\ii{\j-1}
        \node [circle,fill=blue,inner sep=1pt] (p-\j) at (m-\j-1.center){};
        \node [circle,fill=red,inner sep=1pt] (q-\j) at ($(m-\j-1.south west)!0.5!(m-\j-1.south east)$){};
        \ifnum\j<\n
            \pgfmathtruncatemacro\kk{\j+1}
            \draw (m-\j-1.north east)--(p-\j);
            \draw (m-\j-1.south east)--(q-\j);
            \draw (p-\j)--(p-\kk);
            \draw (q-\j)--(q-\kk);
            \ifnum\ii>0
                \draw ($(p-\ii.north west)!0.5!(p-\ii.south west)$)--($(p-\j.north west)!0.5!(p-\j.south west)$);
                \draw ($(q-\ii.south west)!0.5!(q-\ii.north west)$)--($(q-\j.south west)!0.5!(q-\j.north west)$);
            \fi
        \else
            \draw (m-\j-1.north east)--(p-\j);
            \draw (m-\j-1.south east)--(q-\j);
            \ifnum\ii>0
                \draw ($(p-\ii.north west)!0.5!(p-\ii.south west)$)--($(p-\j.north west)!0.5!(p-\j.south west)$);
                \draw ($(q-\ii.south west)!0.5!(q-\ii.north west)$)--($(q-\j.south west)!0.5!(q-\j.north west)$);
            \fi
        \fi
    }

    \node [shade, top color=red!20, bottom color=red!20, opacity=0.4, minimum height=\s, minimum width=1.2*\two*\s, fill] (r) at (m-2-1.center){};
    \node [shade, top color=green!20, bottom color=green!20, opacity=0.4, minimum height=\s, minimum width=1.2*\two*\s, fill] (g) at (m-3-1.center){};
    \node [shade, top color=blue!20, bottom color=blue!20, opacity=0.4, minimum height=\s, minimum width=1.2*\two*\s, fill] (b) at (m-4-1.center){};

    \draw ($(r.south west)!0.5!(r.south east)$)--(r.north east);
    \draw ($(r.north west)!0.5!(r.north east)$)--(r.south east);

    \draw ($(g.south west)!0.5!(g.south east)$)--(g.north east);
    \draw ($(g.north west)!0.5!(g.north east)$)--(g.south east);

    \draw ($(b.south west)!0.5!(b.south east)$)--(b.north east);
    \draw ($(b.north west)!0.5!(b.north east)$)--(b.south east);
\end{tikzpicture}
