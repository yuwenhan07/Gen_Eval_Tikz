\begin{tikzpicture}[scale=.9]
	\tikzstyle{vertex}=[circle, draw, fill=black!60, inner sep=0pt, minimum width=4pt]
	\tikzstyle{edge} = [draw, thick,->,>=stealth]
	\pgfmathsetmacro{\rad}{3}
	\def\ang{-20}
	\foreach \i in {1,...,4}
	{
		\pgfmathsetmacro{\anginc}{\ang+(\i-1)*90}
		\node[vertex,label={\anginc:$\i$}] at (\anginc:\rad) (v\i) {};
		\pgfmathtruncatemacro{\itest}{mod(\i+1,4)+1}
		\pgfmathtruncatemacro{\jtest}{mod(\i,4)+1}
		\pgfmathtruncatemacro{\ktest}{mod(\itest+1,4)+1}
		\pgfmathtruncatemacro{\ltest}{mod(\jtest,4)+1}
		\ifthenelse{\itest=\jtest}
		{
			\draw[edge] (v\i) edge (v\itest);
			\draw[edge] (v\i) edge (v\jtest);
		}
		\fi
	}
	\end{tikzpicture}
