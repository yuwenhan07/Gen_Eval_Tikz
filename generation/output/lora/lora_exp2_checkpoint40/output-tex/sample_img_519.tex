\begin{tikzpicture}[scale=1.4]
  \def\R{1.3}   % radius of the unit disk
  \def\L{2.5}   % length of the thick arrow
  \def\a{60}    % angle between the thick arrows and the complex plane
  \def\h{0.1}   % height of the thick arrow head

  \node[vertex] (v1) at (\a:\R) {$v_1$};
  \node[vertex] (v2) at (-\a:\R) {$v_2$};
  \node[vertex] (v3) at (90:\R) {$v_3$};

  \draw[-latex,thickarr] (v1) -- ++(-\a:\L);
  \draw[-latex,thickarr] (v2) -- ++(\a:\L);
  \draw[-latex,thickarr] (v3) -- ++(-\a:\L);

  \draw[black!40, dashed] (current bounding box.south west)
    rectangle (current bounding box.north east);
  \draw[black!40, dashed] ($(v1)!-1cm!(v2)$) -- ($(v1)!-1cm!(v3)$)
    -- ($(v2)!-1cm!(v3)$) -- cycle;
  \draw[black!40, dashed] ($(v1)!-1cm!(v2)$) -- ($(v1)!-1cm!(v3)$)
    -- ($(v2)!-1cm!(v3)$) -- cycle;

  \node at ($(v1)!0.5!(v2)!0.3!(v3)$) [anchor=west]
    {$\psi_{v_1,v_2}$};

  \node at ($(v1)!1.4!(v2)$) [anchor=south east]
    {$z_2 = \psi_{(v_1,v_2)}(z_1) = e^{i\frac{\pi}{4}} \cdot z_1$};
  \draw[black!80,dashed,-latex] ($(v1)!0.5!(v2)$) ++(-\a:\h)
    -- ++(\a:\L);

  \node at ($(v1)!0.6!(v2)!0.4!(v3)$) [anchor=west]
    {$\psi_{v_1,v_3}$};
  
  \node at ($(v2)!0.8!(v3)$) [anchor=south east]
    {$z_3 = \psi_{(v_2,v_3)}(z_2) = e^{i\frac{\pi}{4}} \cdot z_2$};
  \draw[black!80,dashed,-latex] ($(v2)!0.5!(v3)$) ++(\a:\h)
    -- ++(-\a:\L);

  \node at ($(v3)!0.6!(v2)!0.4!(v1)$) [anchor=west]
    {$\psi_{v_3,v_2}$};
  
  \node at ($(v3)!0.8!(v1)$) [anchor=south east]
    {$z_1 = \psi_{(v_3,v_1)}(z_1') = e^{i\frac{\pi}{4}} \cdot z_1'$};
  \draw[blue,dashed,-latex] ($(v3)!0.5!(v1)$) ++(-\a:\h)
    -- ++(\a:\L);
  

  \node[vertex] (v4) at (current bounding box.south west) {$v_4$};
  \node[vertex] (v5) at ($(v4)!0.8!(v1)$) {$v_5$};

  \draw[black!80,-latex] (v5) -- ++(\a:\L);
  \draw[black!80,-latex] (v4) -- ++(-\a:\L);
  \draw[black!80,-latex] (v5) ++(-\a:\L) -- ++(\a:\L);
  
  \node at ($(v4)!0.5!(v5)$) [anchor=south east]
    {$\psi_{v_4,v_5}$};
  \draw[black!80,dashed,-latex] ($(v4)!0.5!(v5)$) ++(-\a:\h)
    -- ++(\a:\L);
  
\end{tikzpicture}
