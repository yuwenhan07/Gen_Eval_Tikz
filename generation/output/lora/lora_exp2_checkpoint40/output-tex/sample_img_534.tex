\tikzstyle{vertex}=[circle,fill=black!25,minimum size=20pt,inner sep=0pt]

\tikzstyle{leaf}=[circle,fill=yellow,minimum size=8pt,inner sep=0pt]

\def\length{1}
\def\spacing{-0.1}

\def\n{5}
\def\deg{{5,4,4,4,2}}

\foreach \i/\d in {1/\deg[1],2/\deg[2],3/\deg[3],4/\deg[4],5/\deg[5]}
{
    \pgfmathtruncatemacro{\degv}{\deg[\i]-1}
    \coordinate (center-\i) at ({(\i-1)*\length+\spacing*(\i-1)},0);
    \node[vertex] (root-\i) at (center-\i) {$x_\i$};

    \foreach \j in {1,...,\degv}
    {
        \pgfmathtruncatemacro{\x}{\j-1+\spacing*\i}
        \pgfmathtruncatemacro{\y}{\j-1+\spacing*\i+\spacing}
        \pgfmathtruncatemacro{\z}{\j+\spacing*\i}
        \pgfmathtruncatemacro{\w}{\j+\spacing*\i+\spacing}
        \coordinate (tmp--\i-\j) at ({(\z+\w)/2},0);
        \path[draw,-] (root-\i) -- (tmp--\i-\j);
        \foreach \s in {\j,...,\degv}
        {
            \pgfmathtruncatemacro{\a}{\j+\s-1+\spacing*\i}
            \pgfmathtruncatemacro{\b}{\j+\s+\spacing+\spacing*\i}
            \path[draw,-] (tmp--\i-\j) -- (\a,\spacing);
            \path[draw,-] (tmp--\i-\j) -- (\b,-\spacing);
        }
        \node[leaf] (l-\i-\j) at (\x,\spacing) {};
        \node[leaf] (r-\i-\j) at (\z,-\spacing) {};
    }
}
