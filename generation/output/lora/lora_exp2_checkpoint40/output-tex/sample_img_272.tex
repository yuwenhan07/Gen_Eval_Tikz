\begin{tikzpicture}[auto,
dot/.style = {circle, fill=#1, inner sep=1pt, outer sep=0},
dot/.default = black,
every edge/.style = {draw, thick, shorten >=1pt, shorten <=1pt},
every edge quotes/.style = {auto, inner sep=2pt, font=\footnotesize, text=red},
myangle/.style args = {#1/#2}{draw, angle radius=#1, angle eccentricity=#2},
                    ]
\def\diameter{1.5cm}
\foreach \i in {-1, 1}
{
    \ifnum\i=-1
        \pgfmathtruncatemacro{\n}{7}
        \edef\mydir{-}
    \else
        \pgfmathtruncatemacro{\n}{6}
        \edef\mydir{}
    \fi
    \foreach \j in {\n,...,0}
        \coordinate (v\j) at (\mydir\n*90+\j*\diameter/8+\i*\diameter);
}
\draw   (v-1) -- (v-6)
        (v-3) -- (v-7);
%
\path[myangle=1.3cm/1.2]   (v-3) 
            -- node {$\alpha_i$} (v-5);
\path   (v-1) -- coordinate[pos=.5] (v*) (v-3)
        (v-6) -- coordinate[pos=.5] (u*) (v-7);
\path   (u*) ++ (-90:\diameter/4) coordinate (u);
\path   (v*) ++ (-90:\diameter/4) coordinate (v);
\draw   (u) -- node {$\gamma(v_i-\delta)$} (u*)
        (u*) -- node {$\gamma(v_i+\delta)$} (u++);
\draw   (v) -- node {$\gamma(u_i-\delta)$} (v*)
        (v*) -- node {$\gamma(u_i+\delta)$} (v++);
%
\fill (v*) circle[radius=1pt];    % dot between v-3 and v-5
\fill (u*) circle[radius=1pt];    % dot between u-1 and u-6
\foreach \i [count=\ni from 1] in {-1,-1,-1,1,1,1}
    \draw[very thick]   (v\ni) -- (v*\ni) node [pos=.5,above left] {$\gamma(v_i-\theta)$};
\foreach \i [count=\ni from 1] in {-1,1,-1,1,1}
    \draw[very thick]   (u\ni) -- (u*\ni) node [pos=.5,below right] {$\gamma(u_i-\theta)$};
\end{tikzpicture}
