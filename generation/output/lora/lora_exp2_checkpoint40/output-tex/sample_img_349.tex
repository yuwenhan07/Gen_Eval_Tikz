\usepackage{tikz}
\tikzset{
    state/.style={draw,circle,minimum size=1cm},
}
\begin{document}

\begin{tikzpicture}
    \node[state] (a) at (-2,0) {0};
    \node[state] (b) at (2,0) {1};
    \path
      (a) edge [loop above] node[left] {$0$} ()
          edge [bend left=45] node[left] {$0$} (b)
          edge [bend left=45] node[right] {$1$} ()
      (b) edge [loop above] node[left] {$1$} ()
          edge [bend left=45] node[right] {$0$} (a)
          edge [bend left=45] node[right] {$1$} ();
    \path[->]
      (-2,-1) edge [out=-90,in=-90] node[left] {$0$} ()
             edge [out=-90,in=-90] node[left] {$1$} ()
             edge [out=-90,in=-90] node[left] {$0$} (-2,-2);
    \path[->]
      (2,-1) edge [out=-90,in=-90] node[right] {$0$} ()
             edge [out=-90,in=-90] node[right] {$1$} ()
             edge [out=-90,in=-90] node[right] {$0$} (2,-2);
\end{tikzpicture}
