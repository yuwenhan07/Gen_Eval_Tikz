\documentclass[margin=0pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{datavisualization.formats.functions}

\begin{document}
\begin{tikzpicture}[line width=0.4pt]
\draw[help lines] (0,0) grid (7,6);
\datavisualization [school book axes,
                    visualize as smooth line,
                    visualize as scatter={mark size=1.5pt, fill=red},
                    y axis={range=0:7},
                    visualize as scatter={mark size=2pt, fill=white}]
data [format=function] {
var x : interval [-5:5];
func y = (\value x * 3)/(\value x*4) + 2;
};
\end{tikzpicture}
