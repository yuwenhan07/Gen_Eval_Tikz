\tikzset{
    convex polygon/.style={
        shape border uses incircle,
        shape border rotate=0,
        inner sep=0pt,
        minimum width=1cm,
        append after command={
            let \p1=($(\tikzlastnode.corner 1)-(\tikzlastnode.center)$),
                \n1={atan2(\y1,\x1)}
            in 
            (\tikzlastnode.center) foreach \i [count=\j from 1] in {north west,north east,south east,south west}{
                \ifnum\j=1
                    -- (\tikzlastnode.\i)
                \else
                    -- ($(\tikzlastnode.\i)+(\n1:{abs(\p1)})$
                \fi
            }--cycle}}
}
\tikzset{
    every picture/.append style={remember picture},
    every node/.append style={inner sep=0pt},
    decoration={random steps,segment length=3pt,line width=0.2pt},
    mydeco/.style={decorate}}
