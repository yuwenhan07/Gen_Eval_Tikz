\begin{scope}[every node/.style={draw=none,inner sep=0pt}]
  \node (v1) [anchor=south west,xshift=-5ex,yshift=2ex] {vehicle 1};
  \node (v2) [below=of v1] {vehicle 2};
  \node (v3) [below=of v2] {vehicle 3};

  1/port 1,
  2/port 2,
  3/port 2,
  4/port 1}
{
    \pgfmathsetmacro{\n}{\i+1}
    \pgfmathsetmacro{\last}{\i-1}
    \pgfmathsetmacro{\first}{\i}
    \draw[dashed] ([xshift=-3ex]v\n.west) -- ++ (-1,0);
    \draw[dashed] ([xshift=-3ex]v\last.east) -- ++ (-1,0);
    \foreach \t/\col in {1/white,2/lightgray,3/darkgray}{
        \draw[line width=0.8mm,\col] ([xshift=-3ex]v\t.north west) rectangle ++(\n-\first,-3ex);
    }
    \foreach \t/\col in {1/white,2/darkgray,3/lightgray}{
        \draw[line width=0.8mm,\col] ([xshift=-3ex]v\t.south west) rectangle ++(\last-\first,0);
    }
}
