\documentclass[tikz,border=3mm]{standalone}
\usetikzlibrary{knots,arrows.meta,decorations.markings}
\begin{document}
\begin{tikzpicture}[scale=0.8]
  % Define styles
  \tikzset{
    strand/.style={draw=black, line width=0.6pt},
    saddle/.style={draw=red, line width=1pt, double distance=1.2pt},
    crossing/.style={draw=black, line width=0.8pt},
    every path/.append style={decoration={
        markings,
        mark=at position 0.5 with {\arrow{Stealth[scale=0.6]}}
      },
      postaction=decorate
    }
  }
  % Draw initial strands
  \foreach \x in {1,2,3} {
    \draw[strand] (0,\x) -- (1,\x);
  }
  % First saddle move (merge)
  \draw[saddle] (0.5,2) .. controls (0.25,2.5) and (0.25,1.5) .. (0.5,2);
  \node at (0.5,2) {\tiny$\times$};
  % First twist region (positive crossings)
  \begin{knot}[
    flip crossing=1,
    flip crossing=3,
    clip width=4,
    background color=white,
    only when clipping
  ]
    \strand[crossing] (1,1) -- (3,1);
    \strand[crossing] (1,2) -- (3,2);
    \strand[crossing] (1,3) -- (3,3);
    \crossing 1 over 2 (1.5,1.5)
    \crossing 2 over 3 (2.0,2.5)
    \crossing 1 over 2 (2.5,1.5)
  \end{knot}
  % Second twist region (negative crossings)
  \begin{knot}[
    flip crossing=1,
    flip crossing=3,
    clip width=4,
    background color=white,
    only when clipping
  ]
    \strand[crossing] (3,1) -- (5,1);
    \strand[crossing] (3,2) -- (5,2);
    \strand[crossing] (3,3) -- (5,3);
    \crossing 1 under 2 (3.5,1.5)
    \crossing 2 under 3 (4.0,2.5)
    \crossing 1 under 2 (4.5,1.5)
  \end{knot}
  % Second saddle move (split)
  \draw[saddle] (5.5,2) .. controls (5.75,2.5) and (5.75,1.5) .. (5.5,2);
  \node at (5.5,2) {\tiny$\times$};
  % Draw final strands
  \foreach \x in {1,2,3} {
    \draw[strand] (5,\x) -- (6,\x);
  }
  % Annotations
  \node[above] at (0.5,3.2) {Start};
  \node[above] at (5.5,3.2) {End};
  \node at (2,0.5) {First Twist Region};
  \node at (4,0.5) {Second Twist Region};
\end{tikzpicture}
\end{document}
