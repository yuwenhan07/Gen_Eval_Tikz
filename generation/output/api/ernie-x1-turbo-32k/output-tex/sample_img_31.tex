\documentclass[tikz,border=3.14mm]{standalone}
\begin{document}
\begin{tikzpicture}[
    pixel/.style={draw, thick, minimum size=1cm},
    central_pixel/.style={pixel, fill=blue!30},
    neighbor_pixel/.style={pixel, fill=orange!50},
    label/.style={font=\small}
]
% Draw 5x5 pixel grid
\foreach \i in {-2,...,2} {
    \foreach \j in {-2,...,2} {
        \ifnum\i=0 \ifnum\j=0
            \node[central_pixel] at (\i,\j) {};
        \else
            \ifnum\i=1 \ifnum\j=0
                \node[neighbor_pixel] at (\i,\j) {};
            \else\ifnum\i=-1 \ifnum\j=0
                \node[neighbor_pixel] at (\i,\j) {};
            \else\ifnum\i=0 \ifnum\j=1
                \node[neighbor_pixel] at (\i,\j) {};
            \else\ifnum\i=0 \ifnum\j=-1
                \node[neighbor_pixel] at (\i,\j) {};
            \else
                \node[pixel, fill=gray!20] at (\i,\j) {};
            \fi\fi\fi\fi\fi
        \fi
    }
}
% Label central pixel
\node[label] at (0,0) {$p$};
% Label neighbor pixels
\node[label] at (1,0) {$x_p^+$};
\node[label] at (-1,0) {$x_p^-$};
\node[label] at (0,1) {$y_p^+$};
\node[label] at (0,-1) {$y_p^-$};
% Draw arrows indicating problematic pixels
\foreach \i in {-2,...,2} {
    \foreach \j in {-2,...,2} {
        \ifnum\i=0 \ifnum\j=0 \else
            \ifnum\i=1 \ifnum\j=0 \else
                \ifnum\i=-1 \ifnum\j=0 \else
                    \ifnum\i=0 \ifnum\j=1 \else
                        \ifnum\i=0 \ifnum\j=-1 \else
                            \draw[->, orange!70!black, thick, shorten >=2pt, shorten <=2pt] 
                                (0,0) to (\i,\j);
                        \fi\fi
                    \fi\fi
                \fi\fi
            \fi\fi
        \fi
    }
}
% Add caption
\node[align=center, font=\small] at (0,-3) {
    Definition of $x_p^{\pm}$ and $y_p^{\pm}$ for pixel $p$ \\
    in a poorly textured region. Orange points show pixels \\
    that may cause errors in $\Delta x_p$ and $\Delta y_p$ estimation.
};
\end{tikzpicture}
\end{document}
