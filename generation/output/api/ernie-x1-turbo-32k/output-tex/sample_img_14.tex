        \begin{tikzpicture}
            \begin{axis}[
                ybar,
                ymin=0,
                xlabel={Rotation error (rad)},
                ylabel={Frequency},
                title={Rotation Errors ($10^5$ samples)},
                width=\textwidth,
                height=6cm,
                grid=major,
                grid style={dashed, gray!30},
                xtick distance=0.1,

            ] table [
                row sep=\\,
