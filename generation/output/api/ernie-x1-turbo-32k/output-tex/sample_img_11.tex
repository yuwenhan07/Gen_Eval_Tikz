\documentclass[tikz,border=3mm]{standalone}
\usetikzlibrary{arrows.meta, shapes, decorations.markings, calc}
\begin{document}
\begin{tikzpicture}[
    >=Stealth,
    node distance = 12mm and 15mm,
    state/.style = {circle, draw, fill=blue!15, minimum size=8mm,
                    font=\small, label=center:$\rho$},
    amp/.style = {polygon, polygon sides=3, draw, fill=green!15,
                  minimum width=12mm, rotate=90, anchor=south},
    bs/.style = {rectangle, draw, fill=orange!15, minimum width=10mm,
                 minimum height=8mm, font=\small, label=center:BS},
    mode/.style = {circle, draw, fill=red!15, minimum size=6mm,
                  font=\tiny, inner sep=1pt},
    line/.style = {draw, ->, thick},
    every label/.style = {font=\small}
]
    % Initial state
    \node[state] (rho) {};
    
    % Amplifier
    \node[amp, right=of rho] (amp) {Amp};
    
    % Beam Splitter
    \node[bs, right=of amp] (bs) {};
    
    % Thermal noise channels
    \foreach \i in {1,2,3} {
        \coordinate (out\i) at ($(bs.east)+(\i*5mm, -\i*1.5mm)$);
        \node[mode] at (out\i) (m\i) {$\rho_{\i}$};
        \draw[line] (bs.east) -- (m\i);
    }
    
    % Vertical ellipsis for N modes
    \node at ($(bs.east)+(0,-12mm)$) {$\vdots$};
    \node[rotate=90, anchor=south] at ($(bs.east)+(10mm,0)$) {N modes};
    
    % Annotations
    \draw[line] (rho) -- (amp);
    \draw[line] (amp) -- (bs);
    
    % Thermal noise labels
    \foreach \i in {1,2,3} {
        \node[below=1mm of m\i] (n\i) {Thermal noise};
        \draw[line, gray!50, dashed] (m\i.south) -- (n\i.north);
    }
\end{tikzpicture}
\end{document}
