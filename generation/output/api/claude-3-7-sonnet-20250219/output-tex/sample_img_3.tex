  % k's children
  \draw[thick] (k) -- +(-0.75,-0.75) node (kd) {$d$};

  \draw[dotted, thick] (k) -- +(-1.2,-1.2);
  
  % k+p node

  \node[right=0.1cm of kp] {$q_{k+p} = q$};
  
  % k+p's children
