\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{shapes,arrows,positioning}

\begin{document}

\begin{tikzpicture}[
    node distance=1.5cm,
    box/.style={rectangle, draw, fill=black, minimum width=1cm, minimum height=3cm},
    triangle/.style={isosceles triangle, draw, fill=black, minimum width=1.5cm, minimum height=1.5cm, shape border rotate=180},
    arrow/.style={->, >=latex, thick},
    dotted/.style={dotted, thick}
]

% Input state
\node (rho) at (0,0) {$\rho$};

% Amplifier
\node[triangle, right=0.7cm of rho] (amp) {};
\node[below=0.1cm of amp] {Amp};

% Beam splitter
\node[box, right=1cm of amp] (bs) {};
\node[below=0.1cm of bs] {B.S.};

% Output states
\node (rho1) at (6,1) {$\rho_1$};
\node (rhomu) at (6,0) {$\rho_\mu$};
\node (rhoN) at (6,-1) {$\rho_N$};

% Arrows
\draw[arrow] (rho) -- (amp);
\draw[arrow] (amp) -- (bs);
\draw[arrow] (bs.east) -- ++(0.5,0) |- (rho1);
\draw[arrow] (bs.east) -- ++(0.5,0) |- (rhomu);
\draw[arrow] (bs.east) -- ++(0.5,0) |- (rhoN);

% Dotted lines
\draw[dotted] (4.5,0.7) -- (4.5,0.3);
\draw[dotted] (4.5,-0.3) -- (4.5,-0.7);

\end{tikzpicture}

\end{document}
