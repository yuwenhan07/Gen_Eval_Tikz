\documentclass[tikz,border=10pt]{standalone}
\usetikzlibrary{decorations.pathreplacing, arrows.meta, bending}
\usepackage{amssymb}

\begin{document}
\begin{tikzpicture}[
    node distance=1.5cm,
    every node/.style={font=\scriptsize},
    arrow/.style={-{Stealth[bend]}},
    ]
    
    % Define the vertices and edges for the initial graph
    \node (v0) [circle, fill=black, inner sep=0pt, minimum size=4pt] at (0,0) {};
    \node (v1) [above of=v0] [circle, fill=black, inner sep=0pt, minimum size=4pt] {};
    \node (v2) [above of=v1] [circle, fill=black, inner sep=0pt, minimum size=4pt] {};
    
    \node (v0_prime) [right of=v0] [circle, fill=black, inner sep=0pt, minimum size=4pt] {};
    \node (v1_prime) [above of=v0_prime] [circle, fill=black, inner sep=0pt, minimum size=4pt] {};
    \node (v2_prime) [above of=v1_prime] [circle, fill=black, inner sep=0pt, minimum size=4pt] {};
    
    % Draw the edges for the initial graph
    \draw [thick] (v0) -- (v1) node [midway, above, font=\scriptsize] {$v_0$};
    \draw [thick] (v1) -- (v2) node [midway, above, font=\scriptsize] {$v_1$};
    \draw [thick] (v0) -- (v0_prime) node [midway, above, font=\scriptsize] {$v_0'$};
    \draw [thick] (v1) -- (v1_prime) node [midway, above, font=\scriptsize] {$v_1'$};
    \draw [thick, green] (v0) -- (v1_prime);
    \draw [thick, green] (v1) -- (v0_prime);
    \draw [thick, red] (v0_prime) -- (v1_prime);
    \draw [thick, blue] (v2) -- (v2_prime) node [midway, above, font=\scriptsize] {$v_2$};
    \draw [thick, blue] (v2) -- (v1_prime);
    \draw [thick, blue] (v2) -- (v0_prime);
    \draw [thick, red] (v2_prime) -- (v1_prime);
    \draw [thick, red] (v2_prime) -- (v0_prime);
    
    % Define the vertices and edges for the first contraction
    \node (u) [below of=v0_prime, font=\scriptsize] {$u$};
    \node (u_prime) [right of=u] [circle, fill=red, inner sep=0pt, minimum size=6pt] {};
    \draw [thick, red] (u) -- (u_prime) node [midway, above, font=\scriptsize] {$v$};
    
    % Draw the function f1
    \node (f1_label) [right of=v2_prime, font=\scriptsize] {$\tilde{f}_1$};
    \draw [arrow] (v2_prime) -- (f1_label);
    \draw [arrow] (f1_label) -- (u_prime);
    
    % Define the vertices and edges for the intermediate graph
    \node (w0) [below right=1cm and 2cm of u_prime] [circle, fill=black, inner sep=0pt, minimum size=4pt] {};
    \node (w1) [above of=w0] [circle, fill=black, inner sep=0pt, minimum size=4pt] {};
    \node (w0_prime) [right of=w0] [circle, fill=black, inner sep=0pt, minimum size=4pt] {};
    
    \draw [thick, red] (w0) to [out=45,in=135] (w1);
    \draw [thick, red] (w1) to [out=-45,in=-135] (w0);
    \draw [thick, red] (w1) -- (w0_prime);
    \draw [thick, red] (w0) -- (w0_prime);
    \draw [thick, blue] (w0) -- (w0_prime);
    
    % Draw the function f2
    \node (f2_label) [right of=w0_prime, font=\scriptsize] {$\tilde{f}_2$};
    \draw [arrow] (w0_prime) -- (f2_label);
    \draw [arrow] (f2_label) -- (u);
    
    % Define the vertices for the second contraction
    \node (w) [below of=w0, font=\scriptsize] {$w$};
    \node (w_blue) [below right=0.2cm and 0.1cm of w] [circle, fill=blue, inner sep=0pt, minimum size=4pt] {};
    \node (w_red) [above of=w_blue] [circle, fill=red, inner sep=0pt, minimum size=4pt] {};
    \draw [thick, red] (w_red) -- (w_blue);
    \draw [thick, blue] (w_blue) -- (w);
    
    % Draw the function f2 (again, to match the diagram)
    \draw [arrow] (w0) -- (w);
    
    % Define the vertices and edges for the final graph
    \node (u2) [below of=w, font=\scriptsize] {$u$};
    \node (u_final) [right of=u2] [circle, fill=red, inner sep=0pt, minimum size=4pt] {};
    \draw [thick, red] (u2) -- (u_final);
    \draw [thick, red] (u_final) -- (u_final -| w_red) node [midway, below, font=\scriptsize] {$u$};
    
    % Draw the functions f2 and f3
    \node (f2_label2) [right of=w, font=\scriptsize] {$f_2$};
    \node (f3_label) [right of=u2, font=\scriptsize] {$\pi_3$};
    \draw [arrow] (w) -- (f2_label2);
    \draw [arrow] (f2_label2) -- (u2);
    \draw [arrow] (u2) -- (f3_label);
    \draw [arrow] (f3_label) -- (u_final);
    
    % Draw the function f1 (again, to match the diagram)
    \node (f1_label2) [below of=v1, font=\scriptsize] {$f_1$};
    \draw [arrow] (v1) -- (f1_label2);
    \draw [arrow] (f1_label2) -- (u_prime);
    
    % Draw the function pi1 and pi2
    \node (pi1_label) [below of=v0, font=\scriptsize] {$\pi_1$};
    \node (pi2_label) [below of=w0, font=\scriptsize] {$\pi_2$};
    \draw [arrow] (v0) -- (pi1_label);
    \draw [arrow] (pi1_label) -- (u);
    \draw [arrow] (w0) -- (pi2_label);
    \draw [arrow] (pi2_label) -- (w);
    
\end{tikzpicture}
\end{document}
