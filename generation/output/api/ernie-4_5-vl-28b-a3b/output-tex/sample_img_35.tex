\tikzset{%
  every node/.style = {circle, fill = #1, minimum size = 8mm},
  very thick/.style = {line width = 2pt},
  small node/.style = {circle, fill = gray!30, minimum size = 5mm},
  node distance = 7mm and 15mm
}

\begin{tikzpicture}

  % Nodes
  \node[small node] (s1) {\strut$s$};
  \node[small node, right = of s1] (s2) {\strut$\bar{s}$};
  \node[every node, right = of s2] (x1) {\strut$x_1$};
  \node[every node, right = of x1] (x2) {\strut$x_2$};
  \node[every node, right = of x2] (x3) {\strut$x_3$};
  \node[every node, right = of x3] (x4) {\strut$x_4$};
  \node[every node, right = of x4, node distance = 1cm] (xt) {\strut$t$};
  \node[every node, above = of x2] (y1) {\strut$y_1$};
  \node[every node, below = of x2] (y2) {\strut$y_2$};
  \node[every node, above = of x4] (y3) {\strut$y_3$};
  \node[every node, below = of x4] (y4) {\strut$y_4$};
  \node[every node, right = of y4, node distance = 1cm] (yl) {\strut$y_{[c]|}$};
  \node[every node, above = of yl] (yl1) {\strut$x_{[c]|-1}$};
  \node[every node, below = of yl] (yl2) {\strut$x_{[c]|$};
  
  % Long dotted lines
  \draw[dotted] (x3) -- (x4);
  \draw[dotted] (yl1) -- (yl2);

  \draw[very thick, red] (x4) -- (y3);
  \draw[very thick, red] (x4) -- (yl);
  \draw[very thick, red] (yl1) -- (yl);
  \draw[very thick, red] (yl2) -- (xt);
