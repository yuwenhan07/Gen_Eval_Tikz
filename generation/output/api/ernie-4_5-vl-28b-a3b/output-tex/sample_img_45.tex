\documentclass{standalone}
\usepackage{tikz}
\begin{document}

\begin{tikzpicture}[every node/.style={circle, draw, fill=black!50, inner sep=0pt, minimum width=3pt}]
    % First grid
    \begin{scope}[xshift=0cm]
        % Draw vertices and edges
        \foreach \i in {0,1,2,3} {
            \foreach \j in {0,1,2,3} {
                \node (\i-\j) at (1*\i, 1*\j) {};
            }
        }
        \foreach \i in {0,1,2} {
            \foreach \j in {0,1,2,3} {
                \draw (\i.\j) -- (\i+1.\j);
                \draw (\i-\j) -- (\i-\j+1);
            }
        }
        \foreach \i in {0,1,2} {
            \foreach \j in {0,1,2} {
                \draw (\i.\j) -- (\i+1.\j+1);
                \draw (\i.\j) -- (\i+1.\j-1);
            }
        }
        % Mark points A and B and draw L
        \node[fill=red, minimum size=0.1cm] (A) at (0-1) {};
        \node[fill=red, minimum size=0.1cm] (B) at (3-1) {};
        \draw[red, thick] (A) -- (B);
    \end{scope}
    
    % Second grid with curved edges
    \begin{scope}[xshift=6cm]
        % Draw vertices and edges (same as above)
        \foreach \i in {0,1,2,3} {
            \foreach \j in {0,1,2,3} {
                \node (\i-\j) at (1*\i, 1*\j) {};
            }
        }
        \foreach \i in {0,1,2} {
            \foreach \j in {0,1,2,3} {
                \draw (\i.\j) -- (\i+1.\j);
                \draw (\i-\j) -- (\i-\j+1);
            }
        }
        \foreach \i in {0,1,2} {
            \foreach \j in {0,1,2} {
                \draw (\i.\j) -- (\i+1.\j+1);
                \draw (\i.\j) -- (\i+1.\j-1);
            }
        }
        % Mark points A and B and draw L
        \node[fill=red, minimum size=0.1cm] (A) at (0-1) {};
        \node[fill=red, minimum size=0.1cm] (B) at (3-1) {};
        \draw[red, thick] (A) -- (B);
        
        % Draw curved edges from A and B to nearby vertices
        \draw[red, thick, out=30, in=150] (A) to (1-0);
        \draw[red, thick, out=30, in=150] (A) to (1-1);
        \draw[red, thick, out=-30, in=-150] (A) to (2-0);
        \draw[red, thick, out=-30, in=-150] (A) to (2-1);
        \draw[red, thick, out=30, in=150] (B) to (1-3);
        \draw[red, thick, out=30, in=150] (B) to (1-2);
        \draw[red, thick, out=-30, in=-150] (B) to (2-3);
        \draw[red, thick, out=-30, in=-150] (B) to (2-2);
    \end{scope}
\end{tikzpicture}

\end{document}
