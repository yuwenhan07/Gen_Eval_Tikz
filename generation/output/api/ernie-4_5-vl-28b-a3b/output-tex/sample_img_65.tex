\tikzset{
    block/.style={
        rectangle,
        minimum width=1.5cm,
        minimum height=1.5cm,
        outer sep=0pt,
        text centered,
        draw=black,
    },
    matrixstyle/.style={
        matrix of nodes,
        nodes in empty cells,
        column sep=-\pgflinewidth,
        row sep=-\pgflinewidth,
        nodes={
            block,
            anchor=center,
            minimum size=1cm
        }
    }
}
