    \node[red] at (p) {$p$};
    
    % Define the x and y boundaries
    \draw[red, thick] ($(p) + (-1, 1)$) -- ($(p) + (-1, -1)$);
    \draw[red, thick] ($(p) + (1, 1)$) -- ($(p) + (1, -1)$);
    \draw[red, thick] ($(p) + (-1, 1)$) -- ($(p) + (1, 1)$);
    \draw[red, thick] ($(p) + (-1, -1)$) -- ($(p) + (1, -1)$);

    \node[red] at ($(p) + (0, 0.8)$) {$x_p^{-}$};
    \node[red] at ($(p) + (0, -0.8)$) {$x_p^{+}$}; % corrected label placement
    \node[red] at ($(p) + (0.8, 0)$) {$y_p^{+}$}; % corrected label placement
    \node[red] at ($(p) + (-0.8, 0)$) {$y_p^{-}$}; % corrected label placement
