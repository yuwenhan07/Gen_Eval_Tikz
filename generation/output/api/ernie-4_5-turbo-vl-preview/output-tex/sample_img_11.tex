\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{shapes, arrows, positioning}

\begin{document}

\begin{tikzpicture}[
    amp/.style={fill=black, isosceles triangle, draw, minimum width=2cm, minimum height=1.5cm, inner sep=0, shape border rotate=90, anchor=left},
    bs/.style={fill=black, rectangle, draw, minimum width=2cm, minimum height=4cm, inner sep=0},
    >=latex
]
    % Amplifier
    \node[amp] (amp) {Amp};
    \node[right=0.5cm of amp] (bs) [bs] {};
    
    % Input arrow
    \draw[->] (amp.left) -- ++(-1,0) node[above, near start] {$\rho$};
    
    % Output arrows
    \foreach \i in {1,2,...,4} {
        \draw[->, dotted] ([yshift=0.5cm-\i*0.5cm]bs.east) -- ++(0.5,0) node[right] {$\rho_{\i}$};
        \ifnum\i=4
            \draw[->] ([yshift=-1.5cm]bs.east) -- ++(0.5,0) node[right] {$\rho_{N}$};
        \fi
    }
    
    % Label for beam splitter
    \node[right=0.25cm of amp, yshift=0.75cm] {B.S.};
\end{tikzpicture}

\end{document}
