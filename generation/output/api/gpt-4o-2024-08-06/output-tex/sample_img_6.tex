\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{positioning,fit}

\begin{document}

\begin{tikzpicture}[scale=0.8, transform shape]
    % Define styles
    \tikzstyle{vertex}=[circle, draw, inner sep=1pt, minimum size=4pt]
    \tikzstyle{ellipse}=[draw, ellipse, minimum width=20pt, minimum height=60pt]
    
    % Coordinates for left labels
    \foreach \i in {1,2,...,9}
    {
        \node[vertex, label=left:$x_{\i}$] (l\i) at (0, 10-\i) {};
    }
    
    % Vertical groups and their nodes

    {
        \begin{scope}[xshift=\i*1.5cm]
            \node[ellipse, name path=group\i, label=below:{\i$K_9$}] (g\i) at (0, 4.5) {};
            \foreach \j in {1,2,...,9}
            {
                \node[vertex] (v\i\j) at (0, 10-\j) {};
            }
        \end{scope}
    }
    
    % Draw edges
    \foreach \i in {1,2,...,6}
    {
        \foreach \j in {1,2,...,9}
        {
            \draw (l\j) -- (v\i\j);
        }
    }
    
    % Bottom numbers
    \foreach \i in {1,2,...,6}
    {
        \node[vertex, label=below:{\i}] at (\i*1.5cm, 0) {};
    }

    % Draw axes
    \draw[thick] (-0.5, 0) -- (10, 0);
