\begin{tikzpicture}[node distance=1cm, auto, 
    value-vertex/.style={circle, draw=black, fill=green!30, minimum size=10mm},
    variable-vertex/.style={circle, draw=black, fill=red!30, minimum size=10mm},
    constraint-vertex/.style={circle, draw=black, fill=blue!30, minimum size=10mm},
    model-vertex/.style={circle, draw=black, fill=yellow!30, minimum size=10mm},
    operator/.style={circle, draw=black, fill=orange!30, minimum size=10mm},
    tuple/.style={circle, draw=black, fill=orange!10, minimum size=8mm},
    edge/.style={draw, thick},
    operator-box/.style={rectangle, draw=black, fill=gray!10, rounded corners, inner sep=3mm}
]

% Nodes
\node[value-vertex] (v1) {1};
\node[value-vertex] (v2) [right=of v1] {2};

\node[operator] (lhs) [left=of x1] {lhs};
\node[operator] (rhs) [right=of x2] {rhs};
\node[operator] (times1) [below=0.5cm of lhs] {$\times 3$};
\node[operator] (times2) [below=0.5cm of rhs] {$\times 4$};

\node[tuple] (t1) [below=1cm of x1] {$t_1$};
\node[tuple] (t2) [below=1cm of x2] {$t_2$};

\draw[edge] (v2) -- (x1);
\draw[edge] (v2) -- (x2);

\draw[edge] (x1) -- (model);

\draw[edge] (ext) -- (model);

\draw[edge] (lhs) -- (times1);
\draw[edge] (rhs) -- (times2);
\draw[edge] (times1) -- (x1);
\draw[edge] (times2) -- (x2);

\draw[edge] (ineq) -- (lhs);
\draw[edge] (ineq) -- (rhs);

\draw[edge] (t1) -- (x1);
\draw[edge] (t1) -- (ext);
\draw[edge] (t2) -- (x2);
\draw[edge] (t2) -- (ext);

% Operator boxes
\node[operator-box, fit=(lhs) (times1)] {};
\node[operator-box, fit=(rhs) (times2)] {};
