\begin{tikzpicture}[
  >=Stealth,
  font=\small,
  state/.style={draw,rounded corners=2pt,fill=gray!25,inner sep=2pt},
  sol/.style={-{Stealth[length=2.2mm]},line width=0.45pt},
  other/.style={dashed,-{Stealth[length=2.2mm]},line width=0.45pt}
]

% Main nodes (solution nodes are filled gray)
\node[state] (abc) {$\{a,b,c\}$};
\node[state, right=18mm of abc] (ab) {$\{a,b\}$};
\node[state, above right=7mm and 16mm of ab] (bd) {$\{b,d\}$};
\node[state, below right=7mm and 14mm of ab] (b) {$\{b\}$};
\node[state, right=18mm of bd] (cd) {$\{c,d\}$};
\node[state, right=18mm of b] (bcd) {$\{b,c,d\}$};
% place {a,c,d} midway (in y) between {c,d} and {b,c,d}, further to the right
\path (cd) -- (bcd) coordinate[midway] (mid);
\node[state, right=18mm of mid] (acd) {$\{a,c,d\}$};

% External labels (no boxes)
\node[anchor=east] (abd) at ([xshift=-10mm]abc.west) {$\{a,b,d\}$};
\node            (empty) at ($(bd)!0.5!(cd)+(0,13mm)$) {$\emptyset$};
\node[anchor=west] (dlab) at ([xshift=10mm,yshift=6mm]acd.east) {$\{d\}$};
\node[anchor=west] (clab) at ([xshift=10mm,yshift=-6mm]acd.east) {$\{c\}$};
\node[anchor=west] (all)  at ($(b)!0.5!(bcd)+(14mm,-14mm)$) {$\{a,b,c,d\}$};
\node[anchor=west] (bclab) at ([xshift=10mm]bcd.east) {$\{b,c\}$};

% Solid transitions (those used by solutions of length 4)
\draw[sol] (abc) -- (ab);
\draw[sol] (ab) -- (bd);
\draw[sol] (ab) -- (b);
\draw[sol] (bd) -- (cd);
\draw[sol] (b) -- (bcd);
\draw[sol] (cd) -- (acd);
\draw[sol] (bcd) -- (acd);

% Other (dashed) transitions
\draw[other] (abd) -- (abc.west);
\draw[other] (bd) .. controls ($(bd)+(0,11mm)$) and ($(empty)+(-16mm,-2mm)$) .. (empty);
\draw[other] (cd) .. controls ($(cd)+(0,11mm)$) and ($(empty)+(16mm,-2mm)$) .. (empty);
\draw[other] (acd.east) -- (dlab.west);
\draw[other] (acd.east) -- (clab.west);
\draw[other] (b) -- (all.west);
\draw[other] (bcd.east) -- (bclab.west);
% additional dashed internal alternatives (as hinted by the figure)
\draw[other] (abc) .. controls +(0,-12mm) and +(-16mm,8mm) .. (b);
\draw[other] (b) .. controls +(0,12mm) and +(-16mm,-8mm) .. (bd);

\end{tikzpicture}
