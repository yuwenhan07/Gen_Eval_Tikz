\begin{tikzpicture}[>=Stealth, font=\small]
\tikzset{
  mc/.style={draw, rectangle, minimum width=9mm, minimum height=8mm, inner sep=0pt},
  flow/.style={-Stealth, semithick}
}

% Row heights
\def\yt{1.7}
\def\yb{-1.7}

% MC nodes (top: V_L)
\node[mc] (L1) at (0,\yt) {MC};
\node[mc] (L2) at (2.2,\yt) {MC};
\node[mc] (L3) at (4.4,\yt) {MC};
\node        (Ldots) at (6.2,\yt) {$\cdots$};
\node[mc] (L5) at (8.4,\yt) {MC};

% MC nodes (bottom: V_H)
\node[mc] (H1) at (0,\yb) {MC};
\node[mc] (H2) at (2.2,\yb) {MC};
\node[mc] (H3) at (4.4,\yb) {MC};
\node        (Hdots) at (6.2,\yb) {$\cdots$};
\node[mc] (H5) at (8.4,\yb) {MC};

% Enclosing rectangles (nodes)
\draw (-0.95,0.8) rectangle (9.85,2.6);
\draw (-0.95,-2.6) rectangle (9.85,-0.8);

% Left inputs
\draw[flow] (-1.65,\yt) -- (-0.95,\yt);
\node[left] at (-1.65,\yt) {$V_{\mathcal L}$};

\draw[flow] (-1.65,\yb) -- (-0.95,\yb);
\node[left] at (-1.65,\yb) {$V_{\mathcal H}$};

% Horizontal flows inside each node
\draw[flow] (L1.east) -- node[above] {$i_{\mathcal L_1}$} (L2.west);
\draw[flow] (L2.east) -- (L3.west);
\draw[flow] (L3.east) -- (5.6,\yt);
\draw[flow] (6.8,\yt) -- (L5.west);

\draw[flow] (H1.east) -- node[below] {$i_{\mathcal H_1}$} (H2.west);
\draw[flow] (H2.east) -- (H3.west);
\draw[flow] (H3.east) -- (5.6,\yb);
\draw[flow] (6.8,\yb) -- (H5.west);

% Cross-node sharing after each major cycle (MC)
\draw[flow] (L1.east) -- node[pos=.55, left]  {$i_{V_{\mathcal L}2}$} (H2.west);
\draw[flow] (H1.east) -- node[pos=.55, right] {$i_{V_{\mathcal H}2}$} (L2.west);

\draw[flow] (L2.east) -- node[pos=.55, left]  {$i_{V_{\mathcal L}2}$} (H3.west);
\draw[flow] (H2.east) -- node[pos=.55, right] {$i_{V_{\mathcal H}2}$} (L3.west);

% Fusion (sum) and final output
\node[draw, circle, inner sep=1.2pt] (sum) at (11,0) {$\oplus$};
\draw[flow] (L5.east) -- node[pos=.55, right] {$\hat{\imath}_{V_{\mathcal L}}$} (sum);
\draw[flow] (H5.east) -- node[pos=.55, right] {$\hat{\imath}_{V_{\mathcal H}}$} (sum);
\draw[flow] (sum) -- node[above] {$\hat{\imath}$} (12.4,0);
\end{tikzpicture}
