\begin{tikzpicture}[x=1cm,y=0.6cm]
  % Styles
  \tikzset{
    v/.style = {circle, draw, fill=white, inner sep=0pt, minimum size=3pt},
    vf/.style = {circle, draw, fill=black, inner sep=0pt, minimum size=3pt}
  }

  % Axes
  \draw (0,0) -- (13,0);
  \draw (0,0) -- (0,10);

  % y-axis ticks and labels (x_1,...,x_9)
  \foreach \s/\lab in {1/{x_1},2/{x_2},3/{x_3},4/{x_4},5/{x_5},6/{x_6},7/{x_7},8/{x_8},9/{x_9}}{
    \node[anchor=east] at (-0.25,\s) {$\lab$};
    \node[v] at (0.30,\s) {};
  }

  % x-axis ticks and labels (1,...,6)
  \foreach \i in {1,...,6}{
    \pgfmathsetmacro{\X}{2*\i}
    \node[v] at (\X,0) {};
    \node at (\X,-0.55) {$\i$};
  }

  % Copies of K_9 (six columns), their vertices and ellipses
  % Convention: column i has the top (6-i) vertices filled, the rest open.
  \foreach \i in {1,...,6}{
    \pgfmathsetmacro{\X}{2*\i}
    \pgfmathtruncatemacro{\B}{6-\i} % number of filled (top) vertices in column i
    % place the 9 vertices on levels 1..9
    \foreach \s in {1,...,9}{
      \pgfmathtruncatemacro{\isfilled}{ ifthenelse(\s >= 10-\B, 1, 0) }
      \ifnum\isfilled=1
        \node[vf] (n\i-\s) at (\X,\s) {};
      \else
        \node[v]  (n\i-\s) at (\X,\s) {};
      \fi
    }
    % enclose with an ellipse and write the label iK_9
    \node[draw, ellipse, inner sep=7pt, fit=(n\i-1) (n\i-9)] {};
    \node at (\X,0.6) {$\i K_9$};
  }

  % Inter-copy edges induced by the Sierpi≈Ñski product with f(i)=x_i
  % Draw from each filled vertex in column i to an open vertex in column i+1
  % with a consistent downward shift (so curves slope down to the right).
  \foreach \i in {1,...,5}{
    \pgfmathtruncatemacro{\ip}{\i+1}
    \pgfmathtruncatemacro{\Bi}{6-\i}
    \pgfmathtruncatemacro{\Bip}{6-\ip}
    \foreach \k in {0,...,\numexpr\Bi-1\relax}{
      \pgfmathtruncatemacro{\s}{9-\k}                 % filled level in column i
      \pgfmathtruncatemacro{\t}{\s-\Bip}              % mapped open level in column i+1
      \draw (n\i-\s) to[out=-15,in=195] (n\ip-\t);
    }
  }

  % Decorative x-axis caption under the ellipses (optional; remove if not wanted)

\end{tikzpicture}
