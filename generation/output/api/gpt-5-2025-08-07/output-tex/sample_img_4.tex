\begin{tikzpicture}[x=1.2cm,y=1.2cm,>=Stealth, font=\small]

% Helper to draw a tiny Normal-shaped bump over a baseline with right arrow
\newcommand{\smalldist}[1]{%
  % baseline
  \draw[->] ($(#1)+(-1.25,0)$) -- ($(#1)+(1.35,0)$);
  % bell curve
  \draw ($(#1)+(-0.65,0)$)
        .. controls +(0.18,0.85) and +(-0.18,0.85) .. ($(#1)+(0,0)$)
        .. controls +(0.18,-0.85) and +(-0.18,-0.85) .. ($(#1)+(0.65,0)$);
}

% Main anchor points
\coordinate (A) at (0,0);     % heterogeneous preference (aggregate)
\coordinate (B) at (5,0);     % x-related constituent
\coordinate (C) at (9.2,0);   % z-related constituent

% Distributions
\smalldist{A}
\smalldist{B}
\smalldist{C}

% Labels below each small distribution
\node[align=center] at ($(A)+(0,-0.9)$) {Heterogeneous\\preferences};
\node[align=center] at ($(B)+(0,-0.9)$) {$x$-related\\preference};
\node[align=center] at ($(C)+(0,-0.9)$) {$z$-related\\preference};

% Equation symbols between pieces
\node at ($(A)!0.5!(B)+(0,1.0)$) {$=$};
\node at ($(B)!0.5!(C)+(0,1.0)$) {$+$};

% Dots and covariates (to the right of the constituent distributions)
\coordinate (dotx) at ($(B)+(1.25,1.0)$);
\coordinate (dotz) at ($(C)+(1.25,1.0)$);
\node[circle,fill=black,inner sep=1.6pt] at (dotx) {};
\node[circle,fill=black,inner sep=1.6pt] at (dotz) {};
\node at ($(dotx)+(0.6,0)$) {$x_{ijt}$};
\node at ($(dotz)+(0.6,0)$) {$z_{ijt}$};

% Top message and red curved arrows
\node[red,align=center] (toptext) at ($(B)!0.5!(C)+(0,3.3)$)
  {Mixed logit models estimate\\
   constituent mixing distribution parameters};

\draw[red,thick,->]
  ($(toptext.south)+(-1.6,0.05)$) to[out=-80,in=100]
  ($(B)+(0,1.25)$);
\draw[red,thick,->]
  ($(toptext.south)+(1.6,0.05)$)  to[out=-100,in=80]
  ($(C)+(0,1.25)$);

% HAVAN blue sweeping arrows from aggregate to inputs
\draw[blue,thick,->]
  ($(A)+(-1.5,-0.25)$)
  .. controls ($(A)+(0.6,-1.6)$) and ($(B)+(0.4,-1.6)$)
  .. (dotx);
\draw[blue,thick,->]
  ($(A)+(-1.5,-0.25)$)
  .. controls ($(A)+(-0.1,-1.7)$) and ($(C)+(-0.5,-1.7)$)
  .. (dotz);

% Bottom message (blue)
\node[blue,align=center] at ($(B)!0.5!(C)+(0,-2.6)$)
{HAVAN Framework directly relates model inputs\\
 to distributional parameters of aggregate observables-related preference};
