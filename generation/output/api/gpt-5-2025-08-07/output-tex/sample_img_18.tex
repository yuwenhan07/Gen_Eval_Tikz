\begin{tikzpicture}[scale=2.3,>=Latex]
  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  % Left: formal local system before Fourier transform
  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  \begin{scope}
    % Reference circle
    \draw[dotted] (0,0) circle (1);

    % 3-lobed curve
    \draw[blue,domain=0:360,samples=1000]
      plot (\x:{exp(-cos(3*\x+180)/exp(1))});

    % Labels J_1,...,J_6
    \foreach \n in {1,2,...,6}{
      \draw[blue]
        ({120-(\n+1)*60}:{exp(-cos(3*(120-(\n+1)*60)+180)/exp(1))-0.12})
        node {$J_\n$};
    }

    % Orientation arrows and label "1" on each sector boundary
    \foreach \n in {1,2,...,6}{
      \draw[->,blue]
        ({90-\n*60+4}:{1.06*exp(cos(3*(90-(\n+1)*60+4)+180)/exp(1))})
          to
        ({90-\n*60-4}:{1.06*exp(cos(3*(90-(\n+1)*60-4)+180)/exp(1))});
      \draw[blue] ({90-\n*60}:1.2) node {$1$};
    }
  \end{scope}

  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  % Right: formal local system after Fourier transform
  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  \begin{scope}[xshift=3.5cm]
    % Reference circle
    \draw[dotted] (0,0) circle (1);

    % Resulting curve
    \draw[blue,domain=0:720,samples=1000]
      plot (\x:{exp(cos(1.5*\x)/exp(1))});

    % Labels \widehat J_1,...,\widehat J_6
    \foreach \n in {1,2,...,6}{
      \draw[blue]
        ({420-\n*120+60}:{exp(1.4*cos(1.5*(420-\n*120)+90)/exp(1))})
        node {$\widehat J_\n$};
    }

    % Small inner counterflow arrows (with 1 and -1)
    \foreach \n in {0,1,2}{
      \draw[blue,->] ({\n*120+60+5}:0.75) to[bend left] ({\n*120+60-5}:0.85);
      \draw[blue,->] ({\n*120+60+5}:0.85) to[bend left] ({\n*120+60-5}:0.75);
      \draw[blue] ({\n*120+60-8}:0.87) node {\tiny $1$};
      \draw[blue] ({\n*120+60-10}:0.73) node {\tiny $-1$};
    }

    % Partial formal monodromies on the three positive strands
    \draw[blue,->] ({60+5}:1.1) to[bend left] ({60-5}:1.1);
    \draw[blue] ({60}:1.40) node {\tiny $\begin{pmatrix} a & 1 \\ -1 & d \end{pmatrix}$};

    \draw[blue,->] ({180+60+5}:1.1) to[bend left] ({180+60-5}:1.1);
    \draw[blue] ({180+60}:1.40) node {\tiny $\begin{pmatrix} -1 & 0 \\ 0 & 1 \end{pmatrix}$};

    \draw[blue,->] ({-120+60+5}:1.1) to[bend left] ({-120+60-5}:1.1);
    \draw[blue] ({-120+60}:1.40) node {\tiny $\begin{pmatrix} -1 & 0 \\ 0 & 1 \end{pmatrix}$};

    % Basepoint and its label
    \draw ({cos(35)},{sin(35)}) node {\large $\bullet$};
    \draw ({cos(35)+0.10},{sin(35)+0.10}) node {$\widehat{b}$};

    % Strand numbering in green (chosen at \widehat b and continued clockwise)
    \draw[teal] (0:{1.1*exp(-1/exp(1))}) node {\tiny $2$};
    \draw[teal] (0:{0.93*exp(1/exp(1))}) node {\tiny $1$};

    \draw[teal] (-120:{1.1*exp(-1/exp(1))}) node {\tiny $1$};
    \draw[teal] (-120:{0.93*exp(1/exp(1))}) node {\tiny $2$};

    \draw[teal] (120:{1.1*exp(-1/exp(1))}) node {\tiny $2$};
    \draw[teal] (120:{0.93*exp(1/exp(1))}) node {\tiny $1$};

    % Central label K^{-1} indicating the Legendre transform orientation change
    \draw[->] (5:0.28) to[bend left=20] (-25:0.28);
    \node at (-7:0.18) {$K^{-1}$};
  \end{scope}
\end{tikzpicture}
