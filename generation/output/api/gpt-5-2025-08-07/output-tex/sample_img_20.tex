\begin{tikzpicture}[line cap=round,line join=round,x=1cm,y=1cm]

% Global sizes (square window)
\def\W{4.4}   % half–width
\def\H{4.4}   % half–height

% Geometric scales matching the caption
\def\xv{3.1}      % ≈ 1/p from the y–axis (vertical red lines where red first appears)
\def\ycA{1.05}    % ≈ 1/sqrt(p) from the x–axis (first cone vertex)
\def\ycB{1.55}    % second cone vertex (gives darker overlap)
\def\dx{0.65}     % horizontal shift for the inner side cones

% Clip to a square viewport
\begin{scope}
  \clip (-\W,-\H) rectangle (\W,\H);

  % Light red vertical bands (closest vertical lines with red points)
  \fill[red!20] (-\W,-\H) rectangle (-\xv,\H);
  \fill[red!20] (\xv,-\H)  rectangle (\W,\H);

  % Side cones (right and left), two layers for darker intersections
  \fill[red!35] (\xv,0) -- (\W,\H) -- (\W,-\H) -- cycle;
  \fill[red!55] ({\xv-\dx},0) -- (\W,\H) -- (\W,-\H) -- cycle;

  \fill[red!35] (-\xv,0) -- (-\W,\H) -- (-\W,-\H) -- cycle;
  \fill[red!55] ({-\xv+\dx},0) -- (-\W,\H) -- (-\W,-\H) -- cycle;

  % Top and bottom cones with vertices at distance ~ 1/sqrt(p) from x–axis
  \fill[red!35] (0,\ycA) -- (-\W,\H) -- (\W,\H) -- cycle;
  \fill[red!55] (0,\ycB) -- (-\W,\H) -- (\W,\H) -- cycle;

  \fill[red!35] (0,-\ycA) -- (-\W,-\H) -- (\W,-\H) -- cycle;
  \fill[red!55] (0,-\ycB) -- (-\W,-\H) -- (\W,-\H) -- cycle;
\end{scope}

% Axes (origin at the center)
\draw[black,thin] (-\W,0) -- (\W,0);
\draw[black,thin] (0,-\H) -- (0,\H);

% --- Helper to draw the "two-layer blue rectangles" and the horizontal blue line ---
\newcommand{\blockingcluster}[2]{%
  \begin{scope}[shift={(#1,#2)}]
    % Three small staircase stacks (two layers each)
    \foreach \k in {0,0.28,0.56} {
      \draw[blue,line width=0.6pt] (\k,0) rectangle ++(0.22,0.14);
      \draw[blue,line width=0.6pt] (\k+0.07,0.14) rectangle ++(0.22,0.14);
      \draw[blue,line width=0.6pt] (\k+0.14,0.28) rectangle ++(0.22,0.14);
    }
    % Horizontal traversal by the time a red site arrives
    \draw[blue,line width=0.7pt] (-0.04,0.22) -- (0.86,0.22);
  \end{scope}
}

% Place the blue blocking configurations (two per quadrant)
\blockingcluster{-2.5}{ 1.25}
\blockingcluster{-0.8}{ 1.25}
\blockingcluster{ 0.8}{ 1.25}
\blockingcluster{ 2.5}{ 1.25}

\blockingcluster{-2.5}{-1.25}
\blockingcluster{-0.8}{-1.25}
\blockingcluster{ 0.8}{-1.25}
\blockingcluster{ 2.5}{-1.25}

\end{tikzpicture}
