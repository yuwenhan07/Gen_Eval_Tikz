\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta}

\begin{document}

\begin{tikzpicture}[node distance=2cm, on grid, >=Stealth, every node/.style={draw, rectangle, rounded corners, minimum width=1.2cm, minimum height=0.8cm, align=center}]

% Nodes
\node[fill=gray!20] (n1) at (0, 0) {\{a,b,c\}};
\node[fill=gray!20] (n2) [right=of n1] {\{a,b\}};
\node[fill=gray!20] (n3) [right=of n2] {\{b,d\}};
\node (n4) [above right=of n3] {$\emptyset$};
\node[fill=gray!20] (n5) [right=of n3] {\{c,d\}};
\node[fill=gray!20] (n6) [right=of n5] {\{a,c,d\}};
\node[fill=gray!20] (n7) [below=of n3] {\{b\}};
\node[fill=gray!20] (n8) [below=of n5] {\{b,c,d\}};

% Edges
\draw[dashed] (n1) -- node[above] {\{a,b,d\}} (n2);
\draw[-] (n2) -- node[above] {\{b,d\}} (n3);
\draw[dashed] (n3) -- node[left] {\{d\}} (n4);
\draw[-] (n3) -- node[above] {\{c,d\}} (n5);
\draw[dashed] (n5) -- node[above] {\{c\}} (n6);
\draw[-] (n3) -- node[left] {\{b\}} (n7);
\draw[-] (n7) -- node[below] {\{b,c,d\}} (n8);
\draw[dashed] (n8) -- node[right] {\{b,c\}} (n6);

% Loops
\draw[dashed] (n1) edge[loop left] node[left] {\{a,b,d\}} (n1);
\draw[dashed] (n8) edge[loop right] node[right] {\{a,b,c,d\}} (n8);

\end{tikzpicture}

\end{document}
