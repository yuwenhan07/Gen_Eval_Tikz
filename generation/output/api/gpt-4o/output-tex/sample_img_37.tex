\begin{tikzpicture}
    % Define colors
    \definecolor{red}{rgb}{1,0,0}
    
    % Draw the boundary curve \partial \Omega
    \draw[name path=boundary, thick] plot[smooth cycle, tension=0.7] coordinates {(-1.5,0) (0,2) (1.5,0) (0,-2)};
    
    % Draw the x and y axes
    \draw[->, thick] (-2,0) -- (2,0) node[anchor=north] {$x$};
    \draw[->, thick] (0,-2.5) -- (0,2.5) node[anchor=east] {$y$};
    
    % Draw circles and labels
    \draw[name path=circle1, thick] (-1,0) circle (0.5);
    \draw[name path=circle2, thick] (1,0) circle (0.5);
    
    % Intersection points
    \path[name intersections={of=boundary and circle1, by={a,b}}];
    \path[name intersections={of=boundary and circle2, by={c,d}}];
    
    % Label the curves
    \node at (-1,0) {$C_r^-$};
    \node at (1,0) {$C_r^+$};
    
    % Draw K_t^+ and K_t^-
    \draw[red, thick] (a) arc[start angle=180, end angle=360, radius=0.5] node[midway, anchor=north] {$K_t^-$};
    \draw[red, thick] (c) arc[start angle=0, end angle=180, radius=0.5] node[midway, anchor=north] {$K_t^+$};
    
    % Draw L_t
    \draw[red, thick] (b) -- (d) node[midway, anchor=south] {$L_t$};
    
    % Draw labels
    \node at (-1.5,2) {$\partial \Omega$};
    \node at (-1,-0.5) {(-1,0)};
    \node at (1,-0.5) {(1,0)};
    \node at (0,-0.5) {(0,0)};
\end{tikzpicture}
