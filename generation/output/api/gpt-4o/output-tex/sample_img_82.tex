\documentclass[tikz,border=10pt]{standalone}
\usepackage{amsmath}

\begin{document}

\begin{tikzpicture}

% Define epsilon_n
\def\eps{2}

% Define points
\coordinate (0) at (0,0);
\coordinate (1) at (2*\eps,0);
\coordinate (2) at (\eps,{sqrt(3)*\eps});
\coordinate (3) at (\eps,{-sqrt(3)*\eps});
\coordinate (H) at (\eps,0);
\coordinate (G) at (\eps,{sqrt(3)/3*\eps});

% Draw triangle and lines
\draw[thick] (0) -- (1) -- (2) -- cycle;
\draw[dashed] (0) -- (G) -- (1);
\draw (2) -- (G);
\draw[dashed] (H) -- (G);

% Fill vertices
\foreach \point in {0, 1, 2, H, G}
    \fill (\point) circle (2pt);

% Label points
\node[below left] at (0) {$0$};
\node[below right] at (1) {$1$};
\node[above] at (2) {$2$};
\node[below] at (H) {$H$};
\node[right] at (G) {$G$};

% Label sides
\node[left] at ($(0)!.5!(G)$) {$C_1$};
\node[right] at ($(1)!.5!(G)$) {$C_2$};
\node[right] at ($(2)!.5!(G)$) {$C_3$};

\end{tikzpicture}

\end{document}
