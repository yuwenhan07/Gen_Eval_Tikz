\begin{document}

\begin{tikzpicture}[

    variable vertex/.style={circle, draw=black, fill=red!50, minimum size=12mm},
    constraint vertex/.style={circle, draw=black, fill=blue!30, minimum size=12mm},
    model vertex/.style={circle, draw=black, fill=yellow!50, minimum size=12mm},
    operator/.style={circle, draw=black, fill=orange!40, minimum size=8mm},
    tuple/.style={circle, draw=black, fill=orange!30, minimum size=6mm},
    invisible/.style={rectangle, draw=gray!50, fill=gray!20, rounded corners, minimum width=16mm, minimum height=35mm},

\node[value vertex] (v2) at (2,2) {2};
