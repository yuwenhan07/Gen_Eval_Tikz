\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{shapes, positioning}

\begin{document}

\begin{tikzpicture}

% Define car shape
\tikzset{
    car/.pic={
        \node[circle, draw, inner sep=1.5pt] (c) {#1};
        \node[circle, draw, inner sep=1.5pt, above right=0.1cm and 0.2cm of c] (b) {#2};
        \draw[thick] (c.east) -- ++(0.6,0) -- ++(0,-0.2) -- ++(-0.6,0) -- cycle;
        \node[draw, cloud, cloud puffs=8, cloud ignores aspect, minimum width=1cm, minimum height=0.6cm, above=0.4cm of c] (t) {};
        \node[font=\small] at (t) {#2};
    }
}

% Parking spots
\foreach \i in {1,...,7} {
    \draw[thick] (\i,0) -- ++(0.6,0);
    \node[below] at (\i+0.3, 0) {\i};
}

% Cars
\path (1.3,0.3) pic {car=5,2};
\path (2.3,0.3) pic {car=3,3};
\path (3.3,0.3) pic {car=4,3};
\path (4.3,0.3) pic {car=2,5};
\path (5.3,0.3) pic {car=1,7};

\end{tikzpicture}
