\usepackage{pgfplots}

\begin{document}

\begin{tikzpicture}
\begin{axis}[
    axis lines=left,
    xlabel={Input (capital per worker)},
    ylabel={Output per worker},
    xtick=\empty,
    ytick=\empty,
    xmin=0, xmax=10,
    ymin=0, ymax=5,
    xlabel style={at={(axis description cs:0.5,-0.1)},anchor=north},
    ylabel style={at={(axis description cs:-0.1,0.5)},anchor=south,rotate=90},
    width=8cm,
    height=6cm,
    axis x line=bottom,
    axis y line=left,
    clip=false
]

% Lower curve: y = G(N) f(k), assuming a form like a*sqrt(k) for concavity
\addplot[black, domain=0:10, samples=100] {1.5*sqrt(x)};
\node[anchor=west] at (axis cs:8,1.5*sqrt(8)) {$y = G(N) f(k)$};

% Upper curve: y = G(N') f(k), shifted up, e.g., 2*sqrt(k)
\addplot[black, domain=0:10, samples=100] {2*sqrt(x)};
\node[anchor=west] at (axis cs:8,2*sqrt(8)) {$y = G(N') f(k)$};

% Label x-axis with k
\node[anchor=west] at (axis cs:10,0) {$k$};

% Label y-axis with y
\node[anchor=south] at (axis cs:0,5) {$y$};

\end{axis}
